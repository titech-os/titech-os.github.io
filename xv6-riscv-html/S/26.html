 343   int seq, i, n, cc, total;
 344   enum { N=5, SZ=1033 };
 345

 349   }

 351   seq = 0;
 352   if(pid == 0){
 353     close(fds[0]);
 354     for(n = 0; n < N; n++){
 355       for(i = 0; i < SZ; i++)
 356         buf[i] = seq++;
 357       if(write(fds[1], buf, SZ) != SZ){
 358         printf("%s: pipe1 oops 1\n", s);
 359         exit(1);
 360       }
 361     }
 362     exit(0);
 363   } else if(pid > 0){
 364     close(fds[1]);
 365     total = 0;
 366     cc = 1;
 367     while((n = read(fds[0], buf, cc)) > 0){
 368       for(i = 0; i < n; i++){
 369         if((buf[i] & 0xff) != (seq++ & 0xff)){
 370           printf("%s: pipe1 oops 2\n", s);
 371           return;
 372         }
 373       }
 374       total += n;
 375       cc = cc * 2;
 376       if(cc > sizeof(buf))
 377         cc = sizeof(buf);
 378     }
 379     if(total != N * SZ){
 380       printf("%s: pipe1 oops 3 total %d\n", total);
 381       exit(1);
 382     }
 383     close(fds[0]);
 384     wait(&xstatus);
 385     exit(xstatus);
 386   } else {
 387     printf("%s: fork() failed\n", s);

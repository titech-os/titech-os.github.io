  63 // Allocate a zeroed disk block.
  64 static uint
  65 balloc(uint dev)
  66 {
  67   int b, bi, m;
  68   struct buf *bp;
  69 
  70   bp = 0;
  71   for(b = 0; b < sb.size; b += BPB){
  72     bp = bread(dev, BBLOCK(b, sb));
  73     for(bi = 0; bi < BPB && b + bi < sb.size; bi++){
  74       m = 1 << (bi % 8);
  75       if((bp->data[bi/8] & m) == 0){  // Is block free?
  76         bp->data[bi/8] |= m;  // Mark block in use.
  77         log_write(bp);
  78         brelse(bp);
  79         bzero(dev, b + bi);
  80         return b + bi;
  81       }
  82     }
  83     brelse(bp);
  84   }
  85   panic("balloc: out of blocks");
  86 }

  88 // Free a disk block.
  89 static void
  90 bfree(int dev, uint b)
  91 {
  92   struct buf *bp;
  93   int bi, m;
  94 
  95   bp = bread(dev, BBLOCK(b, sb));
  96   bi = b % BPB;
  97   m = 1 << (bi % 8);
  98   if((bp->data[bi/8] & m) == 0)

 105 // Inodes.
 106 //
 107 // An inode describes a single unnamed file.
 108 // The inode disk structure holds metadata: the file's type,
 109 // its size, the number of links referring to it, and the
 110 // list of blocks holding the file's content.
 111 //
 112 // The inodes are laid out sequentially on disk at
 113 // sb.startinode. Each inode has a number, indicating its
 114 // position on the disk.
 115 //
 116 // The kernel keeps a cache of in-use inodes in memory
 117 // to provide a place for synchronizing access
 118 // to inodes used by multiple processes. The cached
 119 // inodes include book-keeping information that is
 120 // not stored on disk: ip->ref and ip->valid.
 121 //
 122 // An inode and its in-memory representation go through a
 123 // sequence of states before they can be used by the
 124 // rest of the file system code.
 125 //
 126 // * Allocation: an inode is allocated if its type (on disk)
 127 //   is non-zero. ialloc() allocates, and iput() frees if
 128 //   the reference and link counts have fallen to zero.
 129 //
 130 // * Referencing in cache: an entry in the inode cache
 131 //   is free if ip->ref is zero. Otherwise ip->ref tracks
 132 //   the number of in-memory pointers to the entry (open
 133 //   files and current directories). iget() finds or
 134 //   creates a cache entry and increments its ref; iput()
 135 //   decrements ref.
 136 //
 137 // * Valid: the information (type, size, &c) in an inode
 138 //   cache entry is only correct when ip->valid is 1.
 139 //   ilock() reads the inode from
 140 //   the disk and sets ip->valid, while iput() clears
 141 //   ip->valid if ip->ref has fallen to zero.
 142 //
 143 // * Locked: file system code may only examine and modify
 144 //   the information in an inode and its content if it
 145 //   has first locked the inode.
 146 //
 147 // Thus a typical sequence is:
 148 //   ip = iget(dev, inum)
 149 //   ilock(ip)
 150 //   ... examine and modify ip->xxx ...
 151 //   iunlock(ip)
 152 //   iput(ip)
 153 //
 154 // ilock() is separate from iget() so that system calls can
 155 // get a long-term reference to an inode (as for an open file)
 156 // and only lock it for short periods (e.g., in read()).
 157 // The separation also helps avoid deadlock and races during
 158 // pathname lookup. iget() increments ip->ref so that the inode
 159 // stays cached and pointers to it remain valid.
 160 //
 161 // Many internal file system functions expect the caller to
 162 // have locked the inodes involved; this lets callers create
 163 // multi-step atomic operations.
 164 //
 165 // The icache.lock spin-lock protects the allocation of icache
 166 // entries. Since ip->ref indicates whether an entry is free,
 167 // and ip->dev and ip->inum indicate which i-node an entry
 168 // holds, one must hold icache.lock while using any of those fields.
 169 //
 170 // An ip->lock sleep-lock protects all ip-> fields other than ref,
 171 // dev, and inum.  One must hold ip->lock in order to
 172 // read or write that inode's ip->valid, ip->size, ip->type, &c.
 173 
 174 struct {
 175   struct spinlock lock;
 176   struct inode inode[NINODE];
 177 } icache;

 192 // Allocate an inode on device dev.
 193 // Mark it as allocated by  giving it type type.
 194 // Returns an unlocked but allocated and referenced inode.
 195 struct inode*
 196 ialloc(uint dev, short type)
 197 {
 198   int inum;
 199   struct buf *bp;
 200   struct dinode *dip;
 201 
 202   for(inum = 1; inum < sb.ninodes; inum++){
 203     bp = bread(dev, IBLOCK(inum, sb));
 204     dip = (struct dinode*)bp->data + inum%IPB;
 205     if(dip->type == 0){  // a free inode
 206       memset(dip, 0, sizeof(*dip));
 207       dip->type = type;
 208       log_write(bp);   // mark it allocated on the disk
 209       brelse(bp);
 210       return iget(dev, inum);
 211     }
 212     brelse(bp);
 213   }
 214   panic("ialloc: no inodes");
 215 }

 217 // Copy a modified in-memory inode to disk.
 218 // Must be called after every change to an ip->xxx field
 219 // that lives on disk, since i-node cache is write-through.
 220 // Caller must hold ip->lock.
 221 void
 222 iupdate(struct inode *ip)
 223 {
 224   struct buf *bp;
 225   struct dinode *dip;
 226 
 227   bp = bread(ip->dev, IBLOCK(ip->inum, sb));
 228   dip = (struct dinode*)bp->data + ip->inum%IPB;
 229   dip->type = ip->type;
 230   dip->major = ip->major;
 231   dip->minor = ip->minor;
 232   dip->nlink = ip->nlink;
 233   dip->size = ip->size;
 234   memmove(dip->addrs, ip->addrs, sizeof(ip->addrs));
 235   log_write(bp);
 236   brelse(bp);
 237 }

 239 // Find the inode with number inum on device dev
 240 // and return the in-memory copy. Does not lock
 241 // the inode and does not read it from disk.
 242 static struct inode*
 243 iget(uint dev, uint inum)
 244 {
 245   struct inode *ip, *empty;
 246 
 247   acquire(&icache.lock);
 248 
 249   // Is the inode already cached?
 250   empty = 0;
 251   for(ip = &icache.inode[0]; ip < &icache.inode[NINODE]; ip++){
 252     if(ip->ref > 0 && ip->dev == dev && ip->inum == inum){
 253       ip->ref++;
 254       release(&icache.lock);
 255       return ip;
 256     }
 257     if(empty == 0 && ip->ref == 0)    // Remember empty slot.
 258       empty = ip;
 259   }
 260 
 261   // Recycle an inode cache entry.
 262   if(empty == 0)
 263     panic("iget: no inodes");
 264 
 265   ip = empty;
 266   ip->dev = dev;
 267   ip->inum = inum;
 268   ip->ref = 1;
 269   ip->valid = 0;
 270   release(&icache.lock);
 271 
 272   return ip;
 273 }

 286 // Lock the given inode.
 287 // Reads the inode from disk if necessary.
 288 void
 289 ilock(struct inode *ip)
 290 {
 291   struct buf *bp;
 292   struct dinode *dip;
 293 
 294   if(ip == 0 || ip->ref < 1)
 295     panic("ilock");
 296 
 297   acquiresleep(&ip->lock);
 298 
 299   if(ip->valid == 0){
 300     bp = bread(ip->dev, IBLOCK(ip->inum, sb));
 301     dip = (struct dinode*)bp->data + ip->inum%IPB;
 302     ip->type = dip->type;
 303     ip->major = dip->major;
 304     ip->minor = dip->minor;
 305     ip->nlink = dip->nlink;
 306     ip->size = dip->size;
 307     memmove(ip->addrs, dip->addrs, sizeof(ip->addrs));
 308     brelse(bp);
 309     ip->valid = 1;
 310     if(ip->type == 0)
 311       panic("ilock: no type");
 312   }
 313 }

 325 // Drop a reference to an in-memory inode.
 326 // If that was the last reference, the inode cache entry can
 327 // be recycled.
 328 // If that was the last reference and the inode has no links
 329 // to it, free the inode (and its content) on disk.
 330 // All calls to iput() must be inside a transaction in
 331 // case it has to free the inode.
 332 void
 333 iput(struct inode *ip)
 334 {
 335   acquire(&icache.lock);
 336 
 337   if(ip->ref == 1 && ip->valid && ip->nlink == 0){
 338     // inode has no links and no other references: truncate and free.
 339 
 340     // ip->ref == 1 means no other process can have ip locked,
 341     // so this acquiresleep() won't block (or deadlock).
 342     acquiresleep(&ip->lock);
 343 
 344     release(&icache.lock);
 345 
 346     itrunc(ip);
 347     ip->type = 0;
 348     iupdate(ip);
 349     ip->valid = 0;
 350 
 351     releasesleep(&ip->lock);
 352 
 353     acquire(&icache.lock);
 354   }
 355 
 356   ip->ref--;
 357   release(&icache.lock);
 358 }

 368 // Inode content
 369 //
 370 // The content (data) associated with each inode is stored
 371 // in blocks on the disk. The first NDIRECT block numbers
 372 // are listed in ip->addrs[].  The next NINDIRECT blocks are
 373 // listed in block ip->addrs[NDIRECT].
 374 
 375 // Return the disk block address of the nth block in inode ip.
 376 // If there is no such block, bmap allocates one.
 377 static uint
 378 bmap(struct inode *ip, uint bn)
 379 {
 380   uint addr, *a;
 381   struct buf *bp;
 382 
 383   if(bn < NDIRECT){
 384     if((addr = ip->addrs[bn]) == 0)
 385       ip->addrs[bn] = addr = balloc(ip->dev);
 386     return addr;
 387   }
 388   bn -= NDIRECT;
 389 
 390   if(bn < NINDIRECT){
 391     // Load indirect block, allocating if necessary.
 392     if((addr = ip->addrs[NDIRECT]) == 0)
 393       ip->addrs[NDIRECT] = addr = balloc(ip->dev);
 394     bp = bread(ip->dev, addr);
 395     a = (uint*)bp->data;
 396     if((addr = a[bn]) == 0){
 397       a[bn] = addr = balloc(ip->dev);
 398       log_write(bp);
 399     }
 400     brelse(bp);
 401     return addr;
 402   }
 403 
 404   panic("bmap: out of range");
 405 }

 407 // Truncate inode (discard contents).
 408 // Caller must hold ip->lock.
 409 void
 410 itrunc(struct inode *ip)
 411 {
 412   int i, j;
 413   struct buf *bp;
 414   uint *a;
 415 
 416   for(i = 0; i < NDIRECT; i++){
 417     if(ip->addrs[i]){
 418       bfree(ip->dev, ip->addrs[i]);
 419       ip->addrs[i] = 0;
 420     }
 421   }
 422 
 423   if(ip->addrs[NDIRECT]){
 424     bp = bread(ip->dev, ip->addrs[NDIRECT]);
 425     a = (uint*)bp->data;
 426     for(j = 0; j < NINDIRECT; j++){
 427       if(a[j])
 428         bfree(ip->dev, a[j]);
 429     }
 430     brelse(bp);
 431     bfree(ip->dev, ip->addrs[NDIRECT]);
 432     ip->addrs[NDIRECT] = 0;
 433   }
 434 
 435   ip->size = 0;
 436   iupdate(ip);
 437 }

 451 // Read data from inode.
 452 // Caller must hold ip->lock.
 453 // If user_dst==1, then dst is a user virtual address;
 454 // otherwise, dst is a kernel address.
 455 int
 456 readi(struct inode *ip, int user_dst, uint64 dst, uint off, uint n)
 457 {
 458   uint tot, m;
 459   struct buf *bp;
 460 
 461   if(off > ip->size || off + n < off)
 462     return 0;
 463   if(off + n > ip->size)
 464     n = ip->size - off;
 465 
 466   for(tot=0; tot<n; tot+=m, off+=m, dst+=m){
 467     bp = bread(ip->dev, bmap(ip, off/BSIZE));
 468     m = min(n - tot, BSIZE - off%BSIZE);
 469     if(either_copyout(user_dst, dst, bp->data + (off % BSIZE), m) == -1) {
 470       brelse(bp);
 471       break;
 472     }
 473     brelse(bp);
 474   }
 475   return tot;
 476 }

 478 // Write data to inode.
 479 // Caller must hold ip->lock.
 480 // If user_src==1, then src is a user virtual address;
 481 // otherwise, src is a kernel address.
 482 int
 483 writei(struct inode *ip, int user_src, uint64 src, uint off, uint n)
 484 {
 485   uint tot, m;
 486   struct buf *bp;
 487 
 488   if(off > ip->size || off + n < off)
 489     return -1;
 490   if(off + n > MAXFILE*BSIZE)
 491     return -1;
 492 
 493   for(tot=0; tot<n; tot+=m, off+=m, src+=m){
 494     bp = bread(ip->dev, bmap(ip, off/BSIZE));
 495     m = min(n - tot, BSIZE - off%BSIZE);
 496     if(either_copyin(bp->data + (off % BSIZE), user_src, src, m) == -1) {
 497       brelse(bp);
 498       break;
 499     }
 500     log_write(bp);
 501     brelse(bp);
 502   }
 503 
 504   if(n > 0){
 505     if(off > ip->size)
 506       ip->size = off;
 507     // write the i-node back to disk even if the size didn't change
 508     // because the loop above might have called bmap() and added a new
 509     // block to ip->addrs[].
 510     iupdate(ip);
 511   }
 512 
 513   return n;
 514 }

 524 // Look for a directory entry in a directory.
 525 // If found, set *poff to byte offset of entry.
 526 struct inode*
 527 dirlookup(struct inode *dp, char *name, uint *poff)
 528 {
 529   uint off, inum;
 530   struct dirent de;
 531 
 532   if(dp->type != T_DIR)
 533     panic("dirlookup not DIR");
 534 
 535   for(off = 0; off < dp->size; off += sizeof(de)){
 536     if(readi(dp, 0, (uint64)&de, off, sizeof(de)) != sizeof(de))
 537       panic("dirlookup read");
 538     if(de.inum == 0)
 539       continue;
 540     if(namecmp(name, de.name) == 0){
 541       // entry matches path element
 542       if(poff)
 543         *poff = off;
 544       inum = de.inum;
 545       return iget(dp->dev, inum);
 546     }
 547   }
 548 
 549   return 0;
 550 }

 552 // Write a new directory entry (name, inum) into the directory dp.
 553 int
 554 dirlink(struct inode *dp, char *name, uint inum)
 555 {
 556   int off;
 557   struct dirent de;
 558   struct inode *ip;
 559 
 560   // Check that name is not present.
 561   if((ip = dirlookup(dp, name, 0)) != 0){
 562     iput(ip);
 563     return -1;
 564   }
 565 
 566   // Look for an empty dirent.
 567   for(off = 0; off < dp->size; off += sizeof(de)){
 568     if(readi(dp, 0, (uint64)&de, off, sizeof(de)) != sizeof(de))
 569       panic("dirlink read");
 570     if(de.inum == 0)
 571       break;
 572   }
 573 
 574   strncpy(de.name, name, DIRSIZ);
 575   de.inum = inum;
 576   if(writei(dp, 0, (uint64)&de, off, sizeof(de)) != sizeof(de))
 577     panic("dirlink");
 578 
 579   return 0;
 580 }

 584 // Copy the next path element from path into name.
 585 // Return a pointer to the element following the copied one.
 586 // The returned path has no leading slashes,
 587 // so the caller can check *path=='\0' to see if the name is the last one.
 588 // If no name to remove, return 0.
 589 //
 590 // Examples:
 591 //   skipelem("a/bb/c", name) = "bb/c", setting name = "a"
 592 //   skipelem("///a//bb", name) = "bb", setting name = "a"
 593 //   skipelem("a", name) = "", setting name = "a"
 594 //   skipelem("", name) = skipelem("////", name) = 0
 595 //
 596 static char*
 597 skipelem(char *path, char *name)
 598 {
 599   char *s;
 600   int len;
 601 
 602   while(*path == '/')
 603     path++;
 604   if(*path == 0)
 605     return 0;
 606   s = path;
 607   while(*path != '/' && *path != 0)
 608     path++;
 609   len = path - s;
 610   if(len >= DIRSIZ)
 611     memmove(name, s, DIRSIZ);
 612   else {
 613     memmove(name, s, len);
 614     name[len] = 0;
 615   }
 616   while(*path == '/')
 617     path++;
 618   return path;
 619 }

 621 // Look up and return the inode for a path name.
 622 // If parent != 0, return the inode for the parent and copy the final
 623 // path element into name, which must have room for DIRSIZ bytes.
 624 // Must be called inside a transaction since it calls iput().
 625 static struct inode*
 626 namex(char *path, int nameiparent, char *name)
 627 {
 628   struct inode *ip, *next;
 629 
 630   if(*path == '/')
 631     ip = iget(ROOTDEV, ROOTINO);
 632   else
 633     ip = idup(myproc()->cwd);
 634 
 635   while((path = skipelem(path, name)) != 0){
 636     ilock(ip);
 637     if(ip->type != T_DIR){
 638       iunlockput(ip);
 639       return 0;
 640     }
 641     if(nameiparent && *path == '\0'){
 642       // Stop one level early.
 643       iunlock(ip);
 644       return ip;
 645     }
 646     if((next = dirlookup(ip, name, 0)) == 0){
 647       iunlockput(ip);
 648       return 0;
 649     }
 650     iunlockput(ip);
 651     ip = next;
 652   }
 653   if(nameiparent){
 654     iput(ip);
 655     return 0;
 656   }
 657   return ip;
 658 }

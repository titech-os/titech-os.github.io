  64 // Allocate a zeroed disk block.
  65 static uint
  66 balloc(uint dev)
  67 {
  68   int b, bi, m;
  69   struct buf *bp;
  70 
  71   bp = 0;
  72   for(b = 0; b < sb.size; b += BPB){
  73     bp = bread(dev, BBLOCK(b, sb));
  74     for(bi = 0; bi < BPB && b + bi < sb.size; bi++){
  75       m = 1 << (bi % 8);
  76       if((bp->data[bi/8] & m) == 0){  // Is block free?
  77         bp->data[bi/8] |= m;  // Mark block in use.
  78         log_write(bp);
  79         brelse(bp);
  80         bzero(dev, b + bi);
  81         return b + bi;
  82       }
  83     }
  84     brelse(bp);
  85   }
  86   panic("balloc: out of blocks");
  87 }

  89 // Free a disk block.
  90 static void
  91 bfree(int dev, uint b)
  92 {
  93   struct buf *bp;
  94   int bi, m;
  95 
  96   bp = bread(dev, BBLOCK(b, sb));
  97   bi = b % BPB;

 106 // Inodes.
 107 //
 108 // An inode describes a single unnamed file.
 109 // The inode disk structure holds metadata: the file's type,
 110 // its size, the number of links referring to it, and the
 111 // list of blocks holding the file's content.
 112 //
 113 // The inodes are laid out sequentially on disk at
 114 // sb.startinode. Each inode has a number, indicating its
 115 // position on the disk.
 116 //
 117 // The kernel keeps a cache of in-use inodes in memory
 118 // to provide a place for synchronizing access
 119 // to inodes used by multiple processes. The cached
 120 // inodes include book-keeping information that is
 121 // not stored on disk: ip->ref and ip->valid.
 122 //
 123 // An inode and its in-memory representation go through a
 124 // sequence of states before they can be used by the
 125 // rest of the file system code.
 126 //
 127 // * Allocation: an inode is allocated if its type (on disk)
 128 //   is non-zero. ialloc() allocates, and iput() frees if
 129 //   the reference and link counts have fallen to zero.
 130 //
 131 // * Referencing in cache: an entry in the inode cache
 132 //   is free if ip->ref is zero. Otherwise ip->ref tracks
 133 //   the number of in-memory pointers to the entry (open
 134 //   files and current directories). iget() finds or
 135 //   creates a cache entry and increments its ref; iput()
 136 //   decrements ref.
 137 //
 138 // * Valid: the information (type, size, &c) in an inode
 139 //   cache entry is only correct when ip->valid is 1.
 140 //   ilock() reads the inode from
 141 //   the disk and sets ip->valid, while iput() clears
 142 //   ip->valid if ip->ref has fallen to zero.
 143 //
 144 // * Locked: file system code may only examine and modify
 145 //   the information in an inode and its content if it
 146 //   has first locked the inode.
 147 //
 148 // Thus a typical sequence is:
 149 //   ip = iget(dev, inum)
 150 //   ilock(ip)
 151 //   ... examine and modify ip->xxx ...
 152 //   iunlock(ip)
 153 //   iput(ip)
 154 //
 155 // ilock() is separate from iget() so that system calls can
 156 // get a long-term reference to an inode (as for an open file)
 157 // and only lock it for short periods (e.g., in read()).
 158 // The separation also helps avoid deadlock and races during
 159 // pathname lookup. iget() increments ip->ref so that the inode
 160 // stays cached and pointers to it remain valid.
 161 //
 162 // Many internal file system functions expect the caller to
 163 // have locked the inodes involved; this lets callers create
 164 // multi-step atomic operations.
 165 //
 166 // The icache.lock spin-lock protects the allocation of icache
 167 // entries. Since ip->ref indicates whether an entry is free,
 168 // and ip->dev and ip->inum indicate which i-node an entry
 169 // holds, one must hold icache.lock while using any of those fields.
 170 //
 171 // An ip->lock sleep-lock protects all ip-> fields other than ref,
 172 // dev, and inum.  One must hold ip->lock in order to
 173 // read or write that inode's ip->valid, ip->size, ip->type, &c.
 174 
 175 struct {
 176   struct spinlock lock;
 177   struct inode inode[NINODE];
 178 } icache;

 193 // Allocate an inode on device dev.
 194 // Mark it as allocated by  giving it type type.
 195 // Returns an unlocked but allocated and referenced inode.
 196 struct inode*
 197 ialloc(uint dev, short type)
 198 {
 199   int inum;
 200   struct buf *bp;
 201   struct dinode *dip;
 202 
 203   for(inum = 1; inum < sb.ninodes; inum++){
 204     bp = bread(dev, IBLOCK(inum, sb));
 205     dip = (struct dinode*)bp->data + inum%IPB;
 206     if(dip->type == 0){  // a free inode
 207       memset(dip, 0, sizeof(*dip));
 208       dip->type = type;
 209       log_write(bp);   // mark it allocated on the disk
 210       brelse(bp);
 211       return iget(dev, inum);
 212     }
 213     brelse(bp);
 214   }
 215   panic("ialloc: no inodes");
 216 }

 218 // Copy a modified in-memory inode to disk.
 219 // Must be called after every change to an ip->xxx field
 220 // that lives on disk, since i-node cache is write-through.
 221 // Caller must hold ip->lock.
 222 void
 223 iupdate(struct inode *ip)
 224 {
 225   struct buf *bp;
 226   struct dinode *dip;
 227 
 228   bp = bread(ip->dev, IBLOCK(ip->inum, sb));
 229   dip = (struct dinode*)bp->data + ip->inum%IPB;
 230   dip->type = ip->type;
 231   dip->major = ip->major;
 232   dip->minor = ip->minor;
 233   dip->nlink = ip->nlink;
 234   dip->size = ip->size;
 235   memmove(dip->addrs, ip->addrs, sizeof(ip->addrs));
 236   log_write(bp);
 237   brelse(bp);
 238 }

 240 // Find the inode with number inum on device dev
 241 // and return the in-memory copy. Does not lock
 242 // the inode and does not read it from disk.
 243 static struct inode*
 244 iget(uint dev, uint inum)
 245 {
 246   struct inode *ip, *empty;
 247 
 248   acquire(&icache.lock);
 249 
 250   // Is the inode already cached?
 251   empty = 0;
 252   for(ip = &icache.inode[0]; ip < &icache.inode[NINODE]; ip++){
 253     if(ip->ref > 0 && ip->dev == dev && ip->inum == inum){
 254       ip->ref++;
 255       release(&icache.lock);
 256       return ip;
 257     }
 258     if(empty == 0 && ip->ref == 0)    // Remember empty slot.
 259       empty = ip;
 260   }
 261 
 262   // Recycle an inode cache entry.
 263   if(empty == 0)
 264     panic("iget: no inodes");
 265 
 266   ip = empty;
 267   ip->dev = dev;
 268   ip->inum = inum;
 269   ip->ref = 1;
 270   ip->valid = 0;
 271   release(&icache.lock);
 272 
 273   return ip;
 274 }

 287 // Lock the given inode.
 288 // Reads the inode from disk if necessary.
 289 void
 290 ilock(struct inode *ip)
 291 {
 292   struct buf *bp;
 293   struct dinode *dip;
 294 
 295   if(ip == 0 || ip->ref < 1)
 296     panic("ilock");
 297 
 298   acquiresleep(&ip->lock);
 299 
 300   if(ip->valid == 0){
 301     bp = bread(ip->dev, IBLOCK(ip->inum, sb));
 302     dip = (struct dinode*)bp->data + ip->inum%IPB;
 303     ip->type = dip->type;
 304     ip->major = dip->major;
 305     ip->minor = dip->minor;
 306     ip->nlink = dip->nlink;
 307     ip->size = dip->size;
 308     memmove(ip->addrs, dip->addrs, sizeof(ip->addrs));
 309     brelse(bp);
 310     ip->valid = 1;
 311     if(ip->type == 0)
 312       panic("ilock: no type");
 313   }
 314 }

 326 // Drop a reference to an in-memory inode.
 327 // If that was the last reference, the inode cache entry can
 328 // be recycled.
 329 // If that was the last reference and the inode has no links
 330 // to it, free the inode (and its content) on disk.
 331 // All calls to iput() must be inside a transaction in
 332 // case it has to free the inode.
 333 void
 334 iput(struct inode *ip)
 335 {
 336   acquire(&icache.lock);
 337 
 338   if(ip->ref == 1 && ip->valid && ip->nlink == 0){
 339     // inode has no links and no other references: truncate and free.
 340 
 341     // ip->ref == 1 means no other process can have ip locked,
 342     // so this acquiresleep() won't block (or deadlock).
 343     acquiresleep(&ip->lock);
 344 
 345     release(&icache.lock);
 346 
 347     itrunc(ip);
 348     ip->type = 0;
 349     iupdate(ip);
 350     ip->valid = 0;
 351 
 352     releasesleep(&ip->lock);
 353 
 354     acquire(&icache.lock);
 355   }
 356 
 357   ip->ref--;
 358   release(&icache.lock);
 359 }

 369 // Inode content
 370 //
 371 // The content (data) associated with each inode is stored
 372 // in blocks on the disk. The first NDIRECT block numbers
 373 // are listed in ip->addrs[].  The next NINDIRECT blocks are
 374 // listed in block ip->addrs[NDIRECT].
 375 
 376 // Return the disk block address of the nth block in inode ip.
 377 // If there is no such block, bmap allocates one.
 378 static uint
 379 bmap(struct inode *ip, uint bn)
 380 {
 381   uint addr, *a;
 382   struct buf *bp;
 383 
 384   if(bn < NDIRECT){
 385     if((addr = ip->addrs[bn]) == 0)
 386       ip->addrs[bn] = addr = balloc(ip->dev);
 387     return addr;
 388   }
 389   bn -= NDIRECT;
 390 
 391   if(bn < NINDIRECT){
 392     // Load indirect block, allocating if necessary.
 393     if((addr = ip->addrs[NDIRECT]) == 0)
 394       ip->addrs[NDIRECT] = addr = balloc(ip->dev);
 395     bp = bread(ip->dev, addr);
 396     a = (uint*)bp->data;
 397     if((addr = a[bn]) == 0){
 398       a[bn] = addr = balloc(ip->dev);
 399       log_write(bp);
 400     }
 401     brelse(bp);
 402     return addr;
 403   }
 404 
 405   panic("bmap: out of range");
 406 }

 408 // Truncate inode (discard contents).
 409 // Only called when the inode has no links
 410 // to it (no directory entries referring to it)
 411 // and has no in-memory reference to it (is
 412 // not an open file or current directory).
 413 static void
 414 itrunc(struct inode *ip)
 415 {
 416   int i, j;
 417   struct buf *bp;
 418   uint *a;
 419 
 420   for(i = 0; i < NDIRECT; i++){
 421     if(ip->addrs[i]){
 422       bfree(ip->dev, ip->addrs[i]);
 423       ip->addrs[i] = 0;
 424     }
 425   }
 426 
 427   if(ip->addrs[NDIRECT]){
 428     bp = bread(ip->dev, ip->addrs[NDIRECT]);
 429     a = (uint*)bp->data;
 430     for(j = 0; j < NINDIRECT; j++){
 431       if(a[j])
 432         bfree(ip->dev, a[j]);
 433     }
 434     brelse(bp);
 435     bfree(ip->dev, ip->addrs[NDIRECT]);
 436     ip->addrs[NDIRECT] = 0;
 437   }
 438 
 439   ip->size = 0;
 440   iupdate(ip);
 441 }

 455 // Read data from inode.
 456 // Caller must hold ip->lock.
 457 // If user_dst==1, then dst is a user virtual address;
 458 // otherwise, dst is a kernel address.
 459 int
 460 readi(struct inode *ip, int user_dst, uint64 dst, uint off, uint n)
 461 {
 462   uint tot, m;
 463   struct buf *bp;
 464 
 465   if(off > ip->size || off + n < off)
 466     return -1;
 467   if(off + n > ip->size)
 468     n = ip->size - off;
 469 
 470   for(tot=0; tot<n; tot+=m, off+=m, dst+=m){
 471     bp = bread(ip->dev, bmap(ip, off/BSIZE));
 472     m = min(n - tot, BSIZE - off%BSIZE);
 473     if(either_copyout(user_dst, dst, bp->data + (off % BSIZE), m) == -1) {
 474       brelse(bp);
 475       break;
 476     }
 477     brelse(bp);
 478   }
 479   return n;
 480 }

 482 // Write data to inode.
 483 // Caller must hold ip->lock.
 484 // If user_src==1, then src is a user virtual address;
 485 // otherwise, src is a kernel address.
 486 int
 487 writei(struct inode *ip, int user_src, uint64 src, uint off, uint n)
 488 {
 489   uint tot, m;
 490   struct buf *bp;
 491 
 492   if(off > ip->size || off + n < off)
 493     return -1;
 494   if(off + n > MAXFILE*BSIZE)
 495     return -1;
 496 
 497   for(tot=0; tot<n; tot+=m, off+=m, src+=m){
 498     bp = bread(ip->dev, bmap(ip, off/BSIZE));
 499     m = min(n - tot, BSIZE - off%BSIZE);
 500     if(either_copyin(bp->data + (off % BSIZE), user_src, src, m) == -1) {
 501       brelse(bp);
 502       break;
 503     }
 504     log_write(bp);
 505     brelse(bp);
 506   }
 507 
 508   if(n > 0){
 509     if(off > ip->size)
 510       ip->size = off;
 511     // write the i-node back to disk even if the size didn't change
 512     // because the loop above might have called bmap() and added a new
 513     // block to ip->addrs[].
 514     iupdate(ip);
 515   }
 516 
 517   return n;
 518 }

 528 // Look for a directory entry in a directory.
 529 // If found, set *poff to byte offset of entry.
 530 struct inode*
 531 dirlookup(struct inode *dp, char *name, uint *poff)
 532 {
 533   uint off, inum;
 534   struct dirent de;
 535 
 536   if(dp->type != T_DIR)
 537     panic("dirlookup not DIR");
 538 
 539   for(off = 0; off < dp->size; off += sizeof(de)){
 540     if(readi(dp, 0, (uint64)&de, off, sizeof(de)) != sizeof(de))
 541       panic("dirlookup read");
 542     if(de.inum == 0)
 543       continue;
 544     if(namecmp(name, de.name) == 0){
 545       // entry matches path element
 546       if(poff)
 547         *poff = off;
 548       inum = de.inum;
 549       return iget(dp->dev, inum);
 550     }
 551   }
 552 
 553   return 0;
 554 }

 556 // Write a new directory entry (name, inum) into the directory dp.
 557 int
 558 dirlink(struct inode *dp, char *name, uint inum)
 559 {
 560   int off;
 561   struct dirent de;
 562   struct inode *ip;
 563 
 564   // Check that name is not present.
 565   if((ip = dirlookup(dp, name, 0)) != 0){
 566     iput(ip);
 567     return -1;
 568   }
 569 
 570   // Look for an empty dirent.
 571   for(off = 0; off < dp->size; off += sizeof(de)){
 572     if(readi(dp, 0, (uint64)&de, off, sizeof(de)) != sizeof(de))
 573       panic("dirlink read");
 574     if(de.inum == 0)
 575       break;
 576   }
 577 
 578   strncpy(de.name, name, DIRSIZ);
 579   de.inum = inum;
 580   if(writei(dp, 0, (uint64)&de, off, sizeof(de)) != sizeof(de))
 581     panic("dirlink");
 582 
 583   return 0;
 584 }

 588 // Copy the next path element from path into name.
 589 // Return a pointer to the element following the copied one.
 590 // The returned path has no leading slashes,
 591 // so the caller can check *path=='\0' to see if the name is the last one.
 592 // If no name to remove, return 0.
 593 //
 594 // Examples:
 595 //   skipelem("a/bb/c", name) = "bb/c", setting name = "a"
 596 //   skipelem("///a//bb", name) = "bb", setting name = "a"
 597 //   skipelem("a", name) = "", setting name = "a"
 598 //   skipelem("", name) = skipelem("////", name) = 0
 599 //
 600 static char*
 601 skipelem(char *path, char *name)
 602 {
 603   char *s;
 604   int len;
 605 
 606   while(*path == '/')
 607     path++;
 608   if(*path == 0)
 609     return 0;
 610   s = path;
 611   while(*path != '/' && *path != 0)
 612     path++;
 613   len = path - s;
 614   if(len >= DIRSIZ)
 615     memmove(name, s, DIRSIZ);
 616   else {
 617     memmove(name, s, len);
 618     name[len] = 0;
 619   }
 620   while(*path == '/')
 621     path++;
 622   return path;
 623 }

 625 // Look up and return the inode for a path name.
 626 // If parent != 0, return the inode for the parent and copy the final
 627 // path element into name, which must have room for DIRSIZ bytes.
 628 // Must be called inside a transaction since it calls iput().
 629 static struct inode*
 630 namex(char *path, int nameiparent, char *name)
 631 {
 632   struct inode *ip, *next;
 633 
 634   if(*path == '/')
 635     ip = iget(ROOTDEV, ROOTINO);
 636   else
 637     ip = idup(myproc()->cwd);
 638 
 639   while((path = skipelem(path, name)) != 0){
 640     ilock(ip);
 641     if(ip->type != T_DIR){
 642       iunlockput(ip);
 643       return 0;
 644     }
 645     if(nameiparent && *path == '\0'){
 646       // Stop one level early.
 647       iunlock(ip);
 648       return ip;
 649     }
 650     if((next = dirlookup(ip, name, 0)) == 0){
 651       iunlockput(ip);
 652       return 0;
 653     }
 654     iunlockput(ip);
 655     ip = next;
 656   }
 657   if(nameiparent){
 658     iput(ip);
 659     return 0;
 660   }
 661   return ip;
 662 }

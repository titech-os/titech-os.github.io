  21 start()
  22 {
  23   // set M Previous Privilege mode to Supervisor, for mret.
  24   unsigned long x = r_mstatus();
  25   x &= ~MSTATUS_MPP_MASK;
  26   x |= MSTATUS_MPP_S;
  27   w_mstatus(x);
  28 
  29   // set M Exception Program Counter to main, for mret.
  30   // requires gcc -mcmodel=medany

  32 
  33   // disable paging for now.
  34   w_satp(0);
  35 
  36   // delegate all interrupts and exceptions to supervisor mode.
  37   w_medeleg(0xffff);
  38   w_mideleg(0xffff);

  40   // ask for clock interrupts.
  41   timerinit();
  42 
  43   // keep each CPU's hartid in its tp register, for cpuid().
  44   int id = r_mhartid();
  45   w_tp(id);

  47   // switch to supervisor mode and jump to main().
  48   asm volatile("mret");

  52 // which arrive at timervec in kernelvec.S,
  53 // which turns them into software interrupts for
  54 // devintr() in trap.c.
  55 void
  56 timerinit()
  57 {
  58   // each CPU has a separate source of timer interrupts.
  59   int id = r_mhartid();
  60 
  61   // ask the CLINT for a timer interrupt.
  62   int interval = 1000000; // cycles; about 1/10th second in qemu.
  63   *(uint64*)CLINT_MTIMECMP(id) = *(uint64*)CLINT_MTIME + interval;
  64 
  65   // prepare information in scratch[] for timervec.
  66   // scratch[0..3] : space for timervec to save registers.
  67   // scratch[4] : address of CLINT MTIMECMP register.
  68   // scratch[5] : desired interval (in cycles) between timer interrupts.
  69   uint64 *scratch = &mscratch0[32 * id];
  70   scratch[4] = CLINT_MTIMECMP(id);
  71   scratch[5] = interval;
  72   w_mscratch((uint64)scratch);

  74   // set the machine-mode trap handler.

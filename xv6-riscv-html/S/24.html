 685 void
 686 pipe1(char *s)
 687 {
 688   int fds[2], pid, xstatus;
 689   int seq, i, n, cc, total;
 690   enum { N=5, SZ=1033 };
 691   
 692   if(pipe(fds) != 0){
 693     printf("%s: pipe() failed\n", s);
 694     exit(1);
 695   }
 696   pid = fork();
 697   seq = 0;
 698   if(pid == 0){
 699     close(fds[0]);
 700     for(n = 0; n < N; n++){
 701       for(i = 0; i < SZ; i++)
 702         buf[i] = seq++;
 703       if(write(fds[1], buf, SZ) != SZ){
 704         printf("%s: pipe1 oops 1\n", s);
 705         exit(1);
 706       }
 707     }
 708     exit(0);
 709   } else if(pid > 0){
 710     close(fds[1]);
 711     total = 0;
 712     cc = 1;
 713     while((n = read(fds[0], buf, cc)) > 0){
 714       for(i = 0; i < n; i++){
 715         if((buf[i] & 0xff) != (seq++ & 0xff)){
 716           printf("%s: pipe1 oops 2\n", s);
 717           return;
 718         }
 719       }
 720       total += n;
 721       cc = cc * 2;
 722       if(cc > sizeof(buf))
 723         cc = sizeof(buf);
 724     }
 725     if(total != N * SZ){
 726       printf("%s: pipe1 oops 3 total %d\n", total);

  19 // Acquire the lock.
  20 // Loops (spins) until the lock is acquired.
  21 void
  22 acquire(struct spinlock *lk)
  23 {
  24   push_off(); // disable interrupts to avoid deadlock.
  25   if(holding(lk))
  26     panic("acquire");
  27 
  28   // On RISC-V, sync_lock_test_and_set turns into an atomic swap:
  29   //   a5 = 1
  30   //   s1 = &lk->locked
  31   //   amoswap.w.aq a5, a5, (s1)
  32   while(__sync_lock_test_and_set(&lk->locked, 1) != 0)
  33     ;
  34 
  35   // Tell the C compiler and the processor to not move loads or stores
  36   // past this point, to ensure that the critical section's memory
  37   // references happen after the lock is acquired.
  38   __sync_synchronize();
  39 
  40   // Record info about lock acquisition for holding() and debugging.
  41   lk->cpu = mycpu();
  42 }

  44 // Release the lock.
  45 void
  46 release(struct spinlock *lk)
  47 {
  48   if(!holding(lk))
  49     panic("release");
  50 
  51   lk->cpu = 0;
  52 
  53   // Tell the C compiler and the CPU to not move loads or stores
  54   // past this point, to ensure that all the stores in the critical
  55   // section are visible to other CPUs before the lock is released.
  56   // On RISC-V, this turns into a fence instruction.
  57   __sync_synchronize();
  58 
  59   // Release the lock, equivalent to lk->locked = 0.
  60   // This code doesn't use a C assignment, since the C standard
  61   // implies that an assignment might be implemented with
  62   // multiple store instructions.
  63   // On RISC-V, sync_lock_release turns into an atomic swap:
  64   //   s1 = &lk->locked
  65   //   amoswap.w zero, zero, (s1)
  66   __sync_lock_release(&lk->locked);
  67 
  68   pop_off();
  69 }

  71 // Check whether this cpu is holding the lock.
  72 int
  73 holding(struct spinlock *lk)
  74 {
  75   int r;
  76   push_off();
  77   r = (lk->locked && lk->cpu == mycpu());
  78   pop_off();
  79   return r;
  80 }

  82 // push_off/pop_off are like intr_off()/intr_on() except that they are matched:
  83 // it takes two pop_off()s to undo two push_off()s.  Also, if interrupts
  84 // are initially off, then push_off, pop_off leaves them off.
  85 
  86 void
  87 push_off(void)
  88 {
  89   int old = intr_get();
  90 
  91   intr_off();
  92   if(mycpu()->noff == 0)
  93     mycpu()->intena = old;
  94   mycpu()->noff += 1;
  95 }

  97 void
  98 pop_off(void)
  99 {
 100   struct cpu *c = mycpu();
 101   if(intr_get())
 102     panic("pop_off - interruptible");
 103   c->noff -= 1;
 104   if(c->noff < 0)
 105     panic("pop_off");
 106   if(c->noff == 0 && c->intena)
 107     intr_on();
 108 }

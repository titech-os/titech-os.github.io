 127 // Load a program segment into pagetable at virtual address va.
 128 // va must be page-aligned
 129 // and the pages from va to va+sz must already be mapped.
 130 // Returns 0 on success, -1 on failure.
 131 static int
 132 loadseg(pagetable_t pagetable, uint64 va, struct inode *ip, uint offset, uint sz)
 133 {
 134   uint i, n;
 135   uint64 pa;
 136 
 137   if((va % PGSIZE) != 0)
 138     panic("loadseg: va must be page aligned");
 139 
 140   for(i = 0; i < sz; i += PGSIZE){
 141     pa = walkaddr(pagetable, va + i);
 142     if(pa == 0)
 143       panic("loadseg: address should exist");
 144     if(sz - i < PGSIZE)
 145       n = sz - i;
 146     else
 147       n = PGSIZE;
 148     if(readi(ip, 0, (uint64)pa, offset+i, n) != n)
 149       return -1;
 150   }
 151   
 152   return 0;
 153 }

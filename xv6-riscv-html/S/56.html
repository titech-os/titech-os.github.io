  11 // virtio mmio control registers, mapped starting at 0x10001000.

  14 #define VIRTIO_MMIO_VERSION             0x004 // version; 1 is legacy

  19 #define VIRTIO_MMIO_GUEST_PAGE_SIZE     0x028 // page size for PFN, write-only

  21 #define VIRTIO_MMIO_QUEUE_NUM_MAX       0x034 // max size of current queue, read-only

  33 // handle an interrupt, exception, or system call from user space.
  34 // called from trampoline.S
  35 //
  36 void
  37 usertrap(void)
  38 {
  39   int which_dev = 0;
  40 
  41   if((r_sstatus() & SSTATUS_SPP) != 0)
  42     panic("usertrap: not from user mode");
  43 
  44   // send interrupts and exceptions to kerneltrap(),
  45   // since we're now in the kernel.
  46   w_stvec((uint64)kernelvec);
  47 
  48   struct proc *p = myproc();
  49   
  50   // save user program counter.
  51   p->tf->epc = r_sepc();
  52   
  53   if(r_scause() == 8){
  54     // system call
  55 
  56     if(p->killed)
  57       exit(-1);
  58 
  59     // sepc points to the ecall instruction,
  60     // but we want to return to the next instruction.
  61     p->tf->epc += 4;
  62 
  63     // an interrupt will change sstatus &c registers,
  64     // so don't enable until done with those registers.
  65     intr_on();
  66 
  67     syscall();
  68   } else if((which_dev = devintr()) != 0){
  69     // ok
  70   } else {
  71     printf("usertrap(): unexpected scause %p pid=%d\n", r_scause(), p->pid);
  72     printf("            sepc=%p stval=%p\n", r_sepc(), r_stval());
  73     p->killed = 1;
  74   }
  75 
  76   if(p->killed)
  77     exit(-1);
  78 
  79   // give up the CPU if this is a timer interrupt.
  80   if(which_dev == 2)
  81     yield();
  82 
  83   usertrapret();
  84 }

  86 //
  87 // return to user space
  88 //
  89 void
  90 usertrapret(void)
  91 {
  92   struct proc *p = myproc();
  93 
  94   // turn off interrupts, since we're switching
  95   // now from kerneltrap() to usertrap().
  96   intr_off();
  97 
  98   // send syscalls, interrupts, and exceptions to trampoline.S
  99   w_stvec(TRAMPOLINE + (uservec - trampoline));
 100 
 101   // set up trapframe values that uservec will need when
 102   // the process next re-enters the kernel.
 103   p->tf->kernel_satp = r_satp();         // kernel page table
 104   p->tf->kernel_sp = p->kstack + PGSIZE; // process's kernel stack
 105   p->tf->kernel_trap = (uint64)usertrap;
 106   p->tf->kernel_hartid = r_tp();         // hartid for cpuid()
 107 
 108   // set up the registers that trampoline.S's sret will use
 109   // to get to user space.
 110   
 111   // set S Previous Privilege mode to User.
 112   unsigned long x = r_sstatus();
 113   x &= ~SSTATUS_SPP; // clear SPP to 0 for user mode
 114   x |= SSTATUS_SPIE; // enable interrupts in user mode
 115   w_sstatus(x);
 116 
 117   // set S Exception Program Counter to the saved user pc.
 118   w_sepc(p->tf->epc);
 119 
 120   // tell trampoline.S the user page table to switch to.
 121   uint64 satp = MAKE_SATP(p->pagetable);
 122 
 123   // jump to trampoline.S at the top of memory, which 
 124   // switches to the user page table, restores user registers,
 125   // and switches to user mode with sret.
 126   uint64 fn = TRAMPOLINE + (userret - trampoline);
 127   ((void (*)(uint64,uint64))fn)(TRAPFRAME, satp);
 128 }

 130 // interrupts and exceptions from kernel code go here via kernelvec,
 131 // on whatever the current kernel stack is.
 132 // must be 4-byte aligned to fit in stvec.
 133 void 
 134 kerneltrap()
 135 {
 136   int which_dev = 0;
 137   uint64 sepc = r_sepc();
 138   uint64 sstatus = r_sstatus();
 139   uint64 scause = r_scause();
 140   
 141   if((sstatus & SSTATUS_SPP) == 0)
 142     panic("kerneltrap: not from supervisor mode");
 143   if(intr_get() != 0)
 144     panic("kerneltrap: interrupts enabled");
 145 
 146   if((which_dev = devintr()) == 0){
 147     printf("scause %p\n", scause);
 148     printf("sepc=%p stval=%p\n", r_sepc(), r_stval());
 149     panic("kerneltrap");
 150   }
 151 
 152   // give up the CPU if this is a timer interrupt.
 153   if(which_dev == 2 && myproc() != 0 && myproc()->state == RUNNING)
 154     yield();
 155 
 156   // the yield() may have caused some traps to occur,
 157   // so restore trap registers for use by kernelvec.S's sepc instruction.
 158   w_sepc(sepc);
 159   w_sstatus(sstatus);
 160 }

 162 void
 163 clockintr()
 164 {
 165   acquire(&tickslock);
 166   ticks++;
 167   wakeup(&ticks);
 168   release(&tickslock);
 169 }

 171 // check if it's an external interrupt or software interrupt,
 172 // and handle it.
 173 // returns 2 if timer interrupt,
 174 // 1 if other device,
 175 // 0 if not recognized.
 176 int
 177 devintr()
 178 {
 179   uint64 scause = r_scause();
 180 
 181   if((scause & 0x8000000000000000L) &&
 182      (scause & 0xff) == 9){
 183     // this is a supervisor external interrupt, via PLIC.
 184 
 185     // irq indicates which device interrupted.
 186     int irq = plic_claim();
 187 
 188     if(irq == UART0_IRQ){
 189       uartintr();
 190     } else if(irq == VIRTIO0_IRQ){
 191       virtio_disk_intr();
 192     }
 193 
 194     plic_complete(irq);
 195     return 1;
 196   } else if(scause == 0x8000000000000001L){
 197     // software interrupt from a machine-mode timer interrupt,
 198     // forwarded by timervec in kernelvec.S.
 199 
 200     if(cpuid() == 0){
 201       clockintr();
 202     }
 203     
 204     // acknowledge the software interrupt by clearing
 205     // the SSIP bit in sip.
 206     w_sip(r_sip() & ~2);
 207 
 208     return 2;
 209   } else {
 210     return 0;
 211   }
 212 }

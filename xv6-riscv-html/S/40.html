  61 // Return the address of the PTE in page table pagetable
  62 // that corresponds to virtual address va.  If alloc!=0,
  63 // create any required page-table pages.
  64 //
  65 // The risc-v Sv39 scheme has three levels of page-table
  66 // pages. A page-table page contains 512 64-bit PTEs.
  67 // A 64-bit virtual address is split into five fields:
  68 //   39..63 -- must be zero.
  69 //   30..38 -- 9 bits of level-2 index.
  70 //   21..39 -- 9 bits of level-1 index.
  71 //   12..20 -- 9 bits of level-0 index.
  72 //    0..12 -- 12 bits of byte offset within the page.
  73 static pte_t *
  74 walk(pagetable_t pagetable, uint64 va, int alloc)
  75 {
  76   if(va >= MAXVA)
  77     panic("walk");
  78 
  79   for(int level = 2; level > 0; level--) {
  80     pte_t *pte = &pagetable[PX(level, va)];
  81     if(*pte & PTE_V) {
  82       pagetable = (pagetable_t)PTE2PA(*pte);
  83     } else {
  84       if(!alloc || (pagetable = (pde_t*)kalloc()) == 0)
  85         return 0;
  86       memset(pagetable, 0, PGSIZE);
  87       *pte = PA2PTE(pagetable) | PTE_V;
  88     }
  89   }
  90   return &pagetable[PX(0, va)];
  91 }

  93 // Look up a virtual address, return the physical address,
  94 // or 0 if not mapped.
  95 // Can only be used to look up user pages.
  96 uint64
  97 walkaddr(pagetable_t pagetable, uint64 va)
  98 {
  99   pte_t *pte;
 100   uint64 pa;
 101 
 102   if(va >= MAXVA)
 103     return 0;
 104 
 105   pte = walk(pagetable, va, 0);
 106   if(pte == 0)
 107     return 0;
 108   if((*pte & PTE_V) == 0)
 109     return 0;
 110   if((*pte & PTE_U) == 0)
 111     return 0;
 112   pa = PTE2PA(*pte);
 113   return pa;
 114 }

 146 // Create PTEs for virtual addresses starting at va that refer to
 147 // physical addresses starting at pa. va and size might not
 148 // be page-aligned. Returns 0 on success, -1 if walk() couldn't
 149 // allocate a needed page-table page.
 150 int
 151 mappages(pagetable_t pagetable, uint64 va, uint64 size, uint64 pa, int perm)
 152 {
 153   uint64 a, last;
 154   pte_t *pte;
 155 
 156   a = PGROUNDDOWN(va);
 157   last = PGROUNDDOWN(va + size - 1);
 158   for(;;){
 159     if((pte = walk(pagetable, a, 1)) == 0)
 160       return -1;
 161     if(*pte & PTE_V)
 162       panic("remap");
 163     *pte = PA2PTE(pa) | perm | PTE_V;
 164     if(a == last)
 165       break;
 166     a += PGSIZE;
 167     pa += PGSIZE;
 168   }
 169   return 0;
 170 }

 172 // Remove mappings from a page table. The mappings in
 173 // the given range must exist. Optionally free the
 174 // physical memory.
 175 void
 176 uvmunmap(pagetable_t pagetable, uint64 va, uint64 size, int do_free)
 177 {
 178   uint64 a, last;
 179   pte_t *pte;
 180   uint64 pa;
 181 
 182   a = PGROUNDDOWN(va);
 183   last = PGROUNDDOWN(va + size - 1);
 184   for(;;){
 185     if((pte = walk(pagetable, a, 0)) == 0)
 186       panic("uvmunmap: walk");
 187     if((*pte & PTE_V) == 0){
 188       printf("va=%p pte=%p\n", a, *pte);
 189       panic("uvmunmap: not mapped");
 190     }
 191     if(PTE_FLAGS(*pte) == PTE_V)
 192       panic("uvmunmap: not a leaf");
 193     if(do_free){
 194       pa = PTE2PA(*pte);
 195       kfree((void*)pa);
 196     }
 197     *pte = 0;
 198     if(a == last)
 199       break;
 200     a += PGSIZE;
 201     pa += PGSIZE;
 202   }
 203 }

 233 // Allocate PTEs and physical memory to grow process from oldsz to
 234 // newsz, which need not be page aligned.  Returns new size or 0 on error.
 235 uint64
 236 uvmalloc(pagetable_t pagetable, uint64 oldsz, uint64 newsz)
 237 {
 238   char *mem;
 239   uint64 a;
 240 
 241   if(newsz < oldsz)
 242     return oldsz;
 243 
 244   oldsz = PGROUNDUP(oldsz);
 245   a = oldsz;
 246   for(; a < newsz; a += PGSIZE){
 247     mem = kalloc();
 248     if(mem == 0){
 249       uvmdealloc(pagetable, a, oldsz);
 250       return 0;
 251     }
 252     memset(mem, 0, PGSIZE);
 253     if(mappages(pagetable, a, PGSIZE, (uint64)mem, PTE_W|PTE_X|PTE_R|PTE_U) != 0){
 254       kfree(mem);
 255       uvmdealloc(pagetable, a, oldsz);
 256       return 0;
 257     }
 258   }
 259   return newsz;
 260 }

 262 // Deallocate user pages to bring the process size from oldsz to
 263 // newsz.  oldsz and newsz need not be page-aligned, nor does newsz
 264 // need to be less than oldsz.  oldsz can be larger than the actual
 265 // process size.  Returns the new process size.
 266 uint64
 267 uvmdealloc(pagetable_t pagetable, uint64 oldsz, uint64 newsz)
 268 {
 269   if(newsz >= oldsz)
 270     return oldsz;
 271 
 272   uint64 newup = PGROUNDUP(newsz);
 273   if(newup < PGROUNDUP(oldsz))
 274     uvmunmap(pagetable, newup, oldsz - newup, 1);
 275 
 276   return newsz;
 277 }

 279 // Recursively free page-table pages.
 280 // All leaf mappings must already have been removed.
 281 static void
 282 freewalk(pagetable_t pagetable)
 283 {
 284   // there are 2^9 = 512 PTEs in a page table.
 285   for(int i = 0; i < 512; i++){
 286     pte_t pte = pagetable[i];
 287     if((pte & PTE_V) && (pte & (PTE_R|PTE_W|PTE_X)) == 0){
 288       // this PTE points to a lower-level page table.
 289       uint64 child = PTE2PA(pte);
 290       freewalk((pagetable_t)child);
 291       pagetable[i] = 0;
 292     } else if(pte & PTE_V){
 293       panic("freewalk: leaf");
 294     }
 295   }
 296   kfree((void*)pagetable);
 297 }

 308 // Given a parent process's page table, copy
 309 // its memory into a child's page table.
 310 // Copies both the page table and the
 311 // physical memory.
 312 // returns 0 on success, -1 on failure.
 313 // frees any allocated pages on failure.
 314 int
 315 uvmcopy(pagetable_t old, pagetable_t new, uint64 sz)
 316 {
 317   pte_t *pte;
 318   uint64 pa, i;
 319   uint flags;
 320   char *mem;
 321 
 322   for(i = 0; i < sz; i += PGSIZE){
 323     if((pte = walk(old, i, 0)) == 0)
 324       panic("uvmcopy: pte should exist");
 325     if((*pte & PTE_V) == 0)
 326       panic("uvmcopy: page not present");
 327     pa = PTE2PA(*pte);
 328     flags = PTE_FLAGS(*pte);
 329     if((mem = kalloc()) == 0)
 330       goto err;
 331     memmove(mem, (char*)pa, PGSIZE);
 332     if(mappages(new, i, PGSIZE, (uint64)mem, flags) != 0){
 333       kfree(mem);
 334       goto err;
 335     }
 336   }
 337   return 0;
 338 
 339  err:
 340   uvmunmap(new, 0, i, 1);
 341   return -1;
 342 }

 357 // Copy from kernel to user.
 358 // Copy len bytes from src to virtual address dstva in a given page table.
 359 // Return 0 on success, -1 on error.
 360 int
 361 copyout(pagetable_t pagetable, uint64 dstva, char *src, uint64 len)
 362 {
 363   uint64 n, va0, pa0;
 364 
 365   while(len > 0){
 366     va0 = PGROUNDDOWN(dstva);
 367     pa0 = walkaddr(pagetable, va0);
 368     if(pa0 == 0)
 369       return -1;
 370     n = PGSIZE - (dstva - va0);
 371     if(n > len)
 372       n = len;
 373     memmove((void *)(pa0 + (dstva - va0)), src, n);
 374 
 375     len -= n;
 376     src += n;
 377     dstva = va0 + PGSIZE;
 378   }
 379   return 0;
 380 }

 407 // Copy a null-terminated string from user to kernel.
 408 // Copy bytes to dst from virtual address srcva in a given page table,
 409 // until a '\0', or max.
 410 // Return 0 on success, -1 on error.
 411 int
 412 copyinstr(pagetable_t pagetable, char *dst, uint64 srcva, uint64 max)
 413 {
 414   uint64 n, va0, pa0;
 415   int got_null = 0;
 416 
 417   while(got_null == 0 && max > 0){
 418     va0 = PGROUNDDOWN(srcva);
 419     pa0 = walkaddr(pagetable, va0);
 420     if(pa0 == 0)
 421       return -1;
 422     n = PGSIZE - (srcva - va0);
 423     if(n > max)
 424       n = max;
 425 
 426     char *p = (char *) (pa0 + (srcva - va0));
 427     while(n > 0){
 428       if(*p == '\0'){
 429         *dst = '\0';
 430         got_null = 1;
 431         break;
 432       } else {
 433         *dst = *p;
 434       }
 435       --n;
 436       --max;
 437       p++;
 438       dst++;
 439     }
 440 
 441     srcva = va0 + PGSIZE;
 442   }
 443   if(got_null){
 444     return 0;
 445   } else {
 446     return -1;
 447   }
 448 }

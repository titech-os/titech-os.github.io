 131 // Load a program segment into pagetable at virtual address va.
 132 // va must be page-aligned
 133 // and the pages from va to va+sz must already be mapped.
 134 // Returns 0 on success, -1 on failure.
 135 static int
 136 loadseg(pagetable_t pagetable, uint64 va, struct inode *ip, uint offset, uint sz)
 137 {
 138   uint i, n;
 139   uint64 pa;
 140 
 141   for(i = 0; i < sz; i += PGSIZE){
 142     pa = walkaddr(pagetable, va + i);
 143     if(pa == 0)
 144       panic("loadseg: address should exist");
 145     if(sz - i < PGSIZE)
 146       n = sz - i;
 147     else
 148       n = PGSIZE;
 149     if(readi(ip, 0, (uint64)pa, offset+i, n) != n)
 150       return -1;
 151   }
 152   
 153   return 0;
 154 }

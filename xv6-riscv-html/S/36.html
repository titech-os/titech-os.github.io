  46   argint(0, &n);

  49 
  50   if(t == SBRK_EAGER || n < 0) {
  51     if(growproc(n) < 0) {
  52       return -1;
  53     }
  54   } else {
  55     // Lazily allocate memory for this process: increase its memory
  56     // size but don't allocate memory. If the processes uses the
  57     // memory, vmfault() will allocate it.

  61   }

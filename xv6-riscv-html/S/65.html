  20 // the address of virtio mmio register r.
  21 #define R(r) ((volatile uint32 *)(VIRTIO0 + (r)))

  23 static struct disk {
  24   // the virtio driver and device mostly communicate through a set of
  25   // structures in RAM. pages[] allocates that memory. pages[] is a
  26   // global (instead of calls to kalloc()) because it must consist of
  27   // two contiguous pages of page-aligned physical memory.
  28   char pages[2*PGSIZE];
  29 
  30   // pages[] is divided into three regions (descriptors, avail, and
  31   // used), as explained in Section 2.6 of the virtio specification
  32   // for the legacy interface.
  33   // https://docs.oasis-open.org/virtio/virtio/v1.1/virtio-v1.1.pdf
  34   
  35   // the first region of pages[] is a set (not a ring) of DMA
  36   // descriptors, with which the driver tells the device where to read
  37   // and write individual disk operations. there are NUM descriptors.
  38   // most commands consist of a "chain" (a linked list) of a couple of
  39   // these descriptors.
  40   // points into pages[].
  41   struct virtq_desc *desc;
  42 
  43   // next is a ring in which the driver writes descriptor numbers
  44   // that the driver would like the device to process.  it only
  45   // includes the head descriptor of each chain. the ring has
  46   // NUM elements.
  47   // points into pages[].
  48   struct virtq_avail *avail;
  49 
  50   // finally a ring in which the device writes descriptor numbers that
  51   // the device has finished processing (just the head of each chain).
  52   // there are NUM used ring entries.
  53   // points into pages[].
  54   struct virtq_used *used;
  55 
  56   // our own book-keeping.
  57   char free[NUM];  // is a descriptor free?
  58   uint16 used_idx; // we've looked this far in used[2..NUM].
  59 
  60   // track info about in-flight operations,
  61   // for use when completion interrupt arrives.
  62   // indexed by first descriptor index of chain.
  63   struct {
  64     struct buf *b;
  65     char status;
  66   } info[NUM];
  67 
  68   // disk command headers.
  69   // one-for-one with descriptors, for convenience.
  70   struct virtio_blk_req ops[NUM];
  71   
  72   struct spinlock vdisk_lock;
  73   
  74 } __attribute__ ((aligned (PGSIZE))) disk;

  76 void
  77 virtio_disk_init(void)
  78 {
  79   uint32 status = 0;
  80 
  81   initlock(&disk.vdisk_lock, "virtio_disk");
  82 
  83   if(*R(VIRTIO_MMIO_MAGIC_VALUE) != 0x74726976 ||
  84      *R(VIRTIO_MMIO_VERSION) != 1 ||
  85      *R(VIRTIO_MMIO_DEVICE_ID) != 2 ||
  86      *R(VIRTIO_MMIO_VENDOR_ID) != 0x554d4551){
  87     panic("could not find virtio disk");
  88   }
  89   
  90   status |= VIRTIO_CONFIG_S_ACKNOWLEDGE;
  91   *R(VIRTIO_MMIO_STATUS) = status;
  92 
  93   status |= VIRTIO_CONFIG_S_DRIVER;
  94   *R(VIRTIO_MMIO_STATUS) = status;
  95 
  96   // negotiate features
  97   uint64 features = *R(VIRTIO_MMIO_DEVICE_FEATURES);
  98   features &= ~(1 << VIRTIO_BLK_F_RO);
  99   features &= ~(1 << VIRTIO_BLK_F_SCSI);
 100   features &= ~(1 << VIRTIO_BLK_F_CONFIG_WCE);
 101   features &= ~(1 << VIRTIO_BLK_F_MQ);
 102   features &= ~(1 << VIRTIO_F_ANY_LAYOUT);
 103   features &= ~(1 << VIRTIO_RING_F_EVENT_IDX);
 104   features &= ~(1 << VIRTIO_RING_F_INDIRECT_DESC);
 105   *R(VIRTIO_MMIO_DRIVER_FEATURES) = features;
 106 
 107   // tell device that feature negotiation is complete.
 108   status |= VIRTIO_CONFIG_S_FEATURES_OK;
 109   *R(VIRTIO_MMIO_STATUS) = status;
 110 
 111   // tell device we're completely ready.
 112   status |= VIRTIO_CONFIG_S_DRIVER_OK;
 113   *R(VIRTIO_MMIO_STATUS) = status;
 114 
 115   *R(VIRTIO_MMIO_GUEST_PAGE_SIZE) = PGSIZE;
 116 
 117   // initialize queue 0.
 118   *R(VIRTIO_MMIO_QUEUE_SEL) = 0;
 119   uint32 max = *R(VIRTIO_MMIO_QUEUE_NUM_MAX);
 120   if(max == 0)
 121     panic("virtio disk has no queue 0");
 122   if(max < NUM)
 123     panic("virtio disk max queue too short");
 124   *R(VIRTIO_MMIO_QUEUE_NUM) = NUM;
 125   memset(disk.pages, 0, sizeof(disk.pages));
 126   *R(VIRTIO_MMIO_QUEUE_PFN) = ((uint64)disk.pages) >> PGSHIFT;
 127 
 128   // desc = pages -- num * virtq_desc
 129   // avail = pages + 0x40 -- 2 * uint16, then num * uint16
 130   // used = pages + 4096 -- 2 * uint16, then num * vRingUsedElem
 131 
 132   disk.desc = (struct virtq_desc *) disk.pages;
 133   disk.avail = (struct virtq_avail *)(disk.pages + NUM*sizeof(struct virtq_desc));
 134   disk.used = (struct virtq_used *) (disk.pages + PGSIZE);
 135 
 136   // all NUM descriptors start out unused.
 137   for(int i = 0; i < NUM; i++)
 138     disk.free[i] = 1;
 139 
 140   // plic.c and trap.c arrange for interrupts from VIRTIO0_IRQ.
 141 }

 143 // find a free descriptor, mark it non-free, return its index.
 144 static int
 145 alloc_desc()
 146 {
 147   for(int i = 0; i < NUM; i++){
 148     if(disk.free[i]){
 149       disk.free[i] = 0;
 150       return i;
 151     }
 152   }
 153   return -1;
 154 }
 155 
 156 // mark a descriptor as free.
 157 static void
 158 free_desc(int i)
 159 {
 160   if(i >= NUM)
 161     panic("free_desc 1");
 162   if(disk.free[i])
 163     panic("free_desc 2");
 164   disk.desc[i].addr = 0;
 165   disk.desc[i].len = 0;
 166   disk.desc[i].flags = 0;
 167   disk.desc[i].next = 0;
 168   disk.free[i] = 1;
 169   wakeup(&disk.free[0]);
 170 }
 171 
 172 // free a chain of descriptors.
 173 static void
 174 free_chain(int i)
 175 {
 176   while(1){
 177     int flag = disk.desc[i].flags;
 178     int nxt = disk.desc[i].next;
 179     free_desc(i);
 180     if(flag & VRING_DESC_F_NEXT)
 181       i = nxt;
 182     else
 183       break;
 184   }
 185 }

 187 // allocate three descriptors (they need not be contiguous).
 188 // disk transfers always use three descriptors.
 189 static int
 190 alloc3_desc(int *idx)
 191 {
 192   for(int i = 0; i < 3; i++){
 193     idx[i] = alloc_desc();
 194     if(idx[i] < 0){
 195       for(int j = 0; j < i; j++)
 196         free_desc(idx[j]);
 197       return -1;
 198     }
 199   }
 200   return 0;
 201 }

 203 void
 204 virtio_disk_rw(struct buf *b, int write)
 205 {
 206   uint64 sector = b->blockno * (BSIZE / 512);
 207 
 208   acquire(&disk.vdisk_lock);
 209 
 210   // the spec's Section 5.2 says that legacy block operations use
 211   // three descriptors: one for type/reserved/sector, one for the
 212   // data, one for a 1-byte status result.
 213 
 214   // allocate the three descriptors.
 215   int idx[3];
 216   while(1){
 217     if(alloc3_desc(idx) == 0) {
 218       break;
 219     }
 220     sleep(&disk.free[0], &disk.vdisk_lock);
 221   }
 222 
 223   // format the three descriptors.
 224   // qemu's virtio-blk.c reads them.
 225 
 226   struct virtio_blk_req *buf0 = &disk.ops[idx[0]];
 227 
 228   if(write)
 229     buf0->type = VIRTIO_BLK_T_OUT; // write the disk
 230   else
 231     buf0->type = VIRTIO_BLK_T_IN; // read the disk
 232   buf0->reserved = 0;
 233   buf0->sector = sector;
 234 
 235   disk.desc[idx[0]].addr = (uint64) buf0;
 236   disk.desc[idx[0]].len = sizeof(struct virtio_blk_req);
 237   disk.desc[idx[0]].flags = VRING_DESC_F_NEXT;
 238   disk.desc[idx[0]].next = idx[1];
 239 
 240   disk.desc[idx[1]].addr = (uint64) b->data;
 241   disk.desc[idx[1]].len = BSIZE;
 242   if(write)
 243     disk.desc[idx[1]].flags = 0; // device reads b->data
 244   else
 245     disk.desc[idx[1]].flags = VRING_DESC_F_WRITE; // device writes b->data
 246   disk.desc[idx[1]].flags |= VRING_DESC_F_NEXT;
 247   disk.desc[idx[1]].next = idx[2];
 248 
 249   disk.info[idx[0]].status = 0xff; // device writes 0 on success
 250   disk.desc[idx[2]].addr = (uint64) &disk.info[idx[0]].status;
 251   disk.desc[idx[2]].len = 1;
 252   disk.desc[idx[2]].flags = VRING_DESC_F_WRITE; // device writes the status
 253   disk.desc[idx[2]].next = 0;
 254 
 255   // record struct buf for virtio_disk_intr().
 256   b->disk = 1;
 257   disk.info[idx[0]].b = b;
 258 
 259   // tell the device the first index in our chain of descriptors.
 260   disk.avail->ring[disk.avail->idx % NUM] = idx[0];
 261 
 262   __sync_synchronize();
 263 
 264   // tell the device another avail ring entry is available.
 265   disk.avail->idx += 1; // not % NUM ...
 266 
 267   __sync_synchronize();
 268 
 269   *R(VIRTIO_MMIO_QUEUE_NOTIFY) = 0; // value is queue number
 270 
 271   // Wait for virtio_disk_intr() to say request has finished.
 272   while(b->disk == 1) {
 273     sleep(b, &disk.vdisk_lock);
 274   }
 275 
 276   disk.info[idx[0]].b = 0;
 277   free_chain(idx[0]);
 278 
 279   release(&disk.vdisk_lock);
 280 }

 282 void
 283 virtio_disk_intr()
 284 {
 285   acquire(&disk.vdisk_lock);
 286 
 287   // the device won't raise another interrupt until we tell it
 288   // we've seen this interrupt, which the following line does.
 289   // this may race with the device writing new entries to
 290   // the "used" ring, in which case we may process the new
 291   // completion entries in this interrupt, and have nothing to do
 292   // in the next interrupt, which is harmless.
 293   *R(VIRTIO_MMIO_INTERRUPT_ACK) = *R(VIRTIO_MMIO_INTERRUPT_STATUS) & 0x3;
 294 
 295   __sync_synchronize();
 296 
 297   // the device increments disk.used->idx when it
 298   // adds an entry to the used ring.
 299 
 300   while(disk.used_idx != disk.used->idx){
 301     __sync_synchronize();
 302     int id = disk.used->ring[disk.used_idx % NUM].id;
 303 
 304     if(disk.info[id].status != 0)
 305       panic("virtio_disk_intr status");
 306 
 307     struct buf *b = disk.info[id].b;
 308     b->disk = 0;   // disk is done with buf
 309     wakeup(b);
 310 
 311     disk.used_idx += 1;
 312   }
 313 
 314   release(&disk.vdisk_lock);
 315 }

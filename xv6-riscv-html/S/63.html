  86 // Look in the process table for an UNUSED proc.
  87 // If found, initialize state required to run in the kernel,
  88 // and return with p->lock held.
  89 // If there are no free procs, return 0.
  90 static struct proc*
  91 allocproc(void)
  92 {
  93   struct proc *p;
  94 
  95   for(p = proc; p < &proc[NPROC]; p++) {
  96     acquire(&p->lock);
  97     if(p->state == UNUSED) {
  98       goto found;
  99     } else {
 100       release(&p->lock);
 101     }
 102   }
 103   return 0;
 104 
 105 found:
 106   p->pid = allocpid();
 107 
 108   // Allocate a trapframe page.
 109   if((p->tf = (struct trapframe *)kalloc()) == 0){
 110     release(&p->lock);
 111     return 0;
 112   }
 113 
 114   // An empty user page table.
 115   p->pagetable = proc_pagetable(p);
 116 
 117   // Set up new context to start executing at forkret,
 118   // which returns to user space.
 119   memset(&p->context, 0, sizeof p->context);
 120   p->context.ra = (uint64)forkret;
 121   p->context.sp = p->kstack + PGSIZE;
 122 
 123   return p;
 124 }

 126 // free a proc structure and the data hanging from it,
 127 // including user pages.
 128 // p->lock must be held.
 129 static void
 130 freeproc(struct proc *p)
 131 {
 132   if(p->tf)
 133     kfree((void*)p->tf);
 134   p->tf = 0;
 135   if(p->pagetable)
 136     proc_freepagetable(p->pagetable, p->sz);
 137   p->pagetable = 0;
 138   p->sz = 0;
 139   p->pid = 0;
 140   p->parent = 0;
 141   p->name[0] = 0;
 142   p->chan = 0;
 143   p->killed = 0;
 144   p->xstate = 0;
 145   p->state = UNUSED;
 146 }

 148 // Create a page table for a given process,
 149 // with no user pages, but with trampoline pages.
 150 pagetable_t
 151 proc_pagetable(struct proc *p)
 152 {
 153   pagetable_t pagetable;
 154 
 155   // An empty page table.
 156   pagetable = uvmcreate();
 157 
 158   // map the trampoline code (for system call return)
 159   // at the highest user virtual address.
 160   // only the supervisor uses it, on the way
 161   // to/from user space, so not PTE_U.
 162   mappages(pagetable, TRAMPOLINE, PGSIZE,
 163            (uint64)trampoline, PTE_R | PTE_X);
 164 
 165   // map the trapframe just below TRAMPOLINE, for trampoline.S.
 166   mappages(pagetable, TRAPFRAME, PGSIZE,
 167            (uint64)(p->tf), PTE_R | PTE_W);
 168 
 169   return pagetable;
 170 }

 195 // Set up first user process.
 196 void
 197 userinit(void)
 198 {
 199   struct proc *p;
 200 
 201   p = allocproc();
 202   initproc = p;
 203   
 204   // allocate one user page and copy init's instructions
 205   // and data into it.
 206   uvminit(p->pagetable, initcode, sizeof(initcode));
 207   p->sz = PGSIZE;
 208 
 209   // prepare for the very first "return" from kernel to user.
 210   p->tf->epc = 0;      // user program counter
 211   p->tf->sp = PGSIZE;  // user stack pointer
 212 
 213   safestrcpy(p->name, "initcode", sizeof(p->name));
 214   p->cwd = namei("/");
 215 
 216   p->state = RUNNABLE;
 217 
 218   release(&p->lock);
 219 }

 221 // Grow or shrink user memory by n bytes.
 222 // Return 0 on success, -1 on failure.
 223 int
 224 growproc(int n)
 225 {
 226   uint sz;
 227   struct proc *p = myproc();
 228 
 229   sz = p->sz;
 230   if(n > 0){
 231     if((sz = uvmalloc(p->pagetable, sz, sz + n)) == 0) {
 232       return -1;
 233     }
 234   } else if(n < 0){
 235     if((sz = uvmdealloc(p->pagetable, sz, sz + n)) == 0) {
 236       return -1;
 237     }
 238   }
 239   p->sz = sz;
 240   return 0;
 241 }

 243 // Create a new process, copying the parent.
 244 // Sets up child kernel stack to return as if from fork() system call.
 245 int
 246 fork(void)
 247 {
 248   int i, pid;
 249   struct proc *np;
 250   struct proc *p = myproc();
 251 
 252   // Allocate process.
 253   if((np = allocproc()) == 0){
 254     return -1;
 255   }
 256 
 257   // Copy user memory from parent to child.
 258   if(uvmcopy(p->pagetable, np->pagetable, p->sz) < 0){
 259     freeproc(np);
 260     release(&np->lock);
 261     return -1;
 262   }
 263   np->sz = p->sz;
 264 
 265   np->parent = p;
 266 
 267   // copy saved user registers.
 268   *(np->tf) = *(p->tf);
 269 
 270   // Cause fork to return 0 in the child.
 271   np->tf->a0 = 0;
 272 
 273   // increment reference counts on open file descriptors.
 274   for(i = 0; i < NOFILE; i++)
 275     if(p->ofile[i])
 276       np->ofile[i] = filedup(p->ofile[i]);
 277   np->cwd = idup(p->cwd);
 278 
 279   safestrcpy(np->name, p->name, sizeof(p->name));
 280 
 281   pid = np->pid;
 282 
 283   np->state = RUNNABLE;
 284 
 285   release(&np->lock);
 286 
 287   return pid;
 288 }

 290 // Pass p's abandoned children to init.
 291 // Caller must hold p->lock and parent->lock.
 292 void
 293 reparent(struct proc *p, struct proc *parent) {
 294   struct proc *pp;
 295   int child_of_init = (p->parent == initproc);
 296 
 297   for(pp = proc; pp < &proc[NPROC]; pp++){
 298     // this code uses pp->parent without holding pp->lock.
 299     // acquiring the lock first could cause a deadlock
 300     // if pp or a child of pp were also in exit()
 301     // and about to try to lock p.
 302     if(pp->parent == p){
 303       // pp->parent can't change between the check and the acquire()
 304       // because only the parent changes it, and we're the parent.
 305       acquire(&pp->lock);
 306       pp->parent = initproc;
 307       if(pp->state == ZOMBIE) {
 308         if(!child_of_init)
 309           acquire(&initproc->lock);
 310         wakeup1(initproc);
 311         if(!child_of_init)
 312           release(&initproc->lock);
 313       }
 314       release(&pp->lock);
 315     }
 316   }
 317 }

 319 // Exit the current process.  Does not return.
 320 // An exited process remains in the zombie state
 321 // until its parent calls wait().
 322 void
 323 exit(int status)
 324 {
 325   struct proc *p = myproc();
 326 
 327   if(p == initproc)
 328     panic("init exiting");
 329 
 330   // Close all open files.
 331   for(int fd = 0; fd < NOFILE; fd++){
 332     if(p->ofile[fd]){
 333       struct file *f = p->ofile[fd];
 334       fileclose(f);
 335       p->ofile[fd] = 0;
 336     }
 337   }
 338 
 339   begin_op();
 340   iput(p->cwd);
 341   end_op();
 342   p->cwd = 0;
 343 
 344   acquire(&p->parent->lock);
 345     
 346   acquire(&p->lock);
 347 
 348   // Give any children to init.
 349   reparent(p, p->parent);
 350 
 351   // Parent might be sleeping in wait().
 352   wakeup1(p->parent);
 353 
 354   p->xstate = status;
 355   p->state = ZOMBIE;
 356 
 357   release(&p->parent->lock);
 358 
 359   // Jump into the scheduler, never to return.
 360   sched();
 361   panic("zombie exit");
 362 }

 364 // Wait for a child process to exit and return its pid.
 365 // Return -1 if this process has no children.
 366 int
 367 wait(uint64 addr)
 368 {
 369   struct proc *np;
 370   int havekids, pid;
 371   struct proc *p = myproc();
 372 
 373   // hold p->lock for the whole time to avoid lost
 374   // wakeups from a child's exit().
 375   acquire(&p->lock);
 376 
 377   for(;;){
 378     // Scan through table looking for exited children.
 379     havekids = 0;
 380     for(np = proc; np < &proc[NPROC]; np++){
 381       // this code uses np->parent without holding np->lock.
 382       // acquiring the lock first would cause a deadlock,
 383       // since np might be an ancestor, and we already hold p->lock.
 384       if(np->parent == p){
 385         // np->parent can't change between the check and the acquire()
 386         // because only the parent changes it, and we're the parent.
 387         acquire(&np->lock);
 388         havekids = 1;
 389         if(np->state == ZOMBIE){
 390           // Found one.
 391           pid = np->pid;
 392           if(addr != 0 && copyout(p->pagetable, addr, (char *)&np->xstate,
 393                                   sizeof(np->xstate)) < 0) {
 394             release(&np->lock);
 395             release(&p->lock);
 396             return -1;
 397           }
 398           freeproc(np);
 399           release(&np->lock);
 400           release(&p->lock);
 401           return pid;
 402         }
 403         release(&np->lock);
 404       }
 405     }
 406 
 407     // No point waiting if we don't have any children.
 408     if(!havekids || p->killed){
 409       release(&p->lock);
 410       return -1;
 411     }
 412     
 413     // Wait for a child to exit.
 414     sleep(p, &p->lock);  //DOC: wait-sleep
 415   }
 416 }

 418 // Per-CPU process scheduler.
 419 // Each CPU calls scheduler() after setting itself up.
 420 // Scheduler never returns.  It loops, doing:
 421 //  - choose a process to run.
 422 //  - swtch to start running that process.
 423 //  - eventually that process transfers control
 424 //    via swtch back to the scheduler.
 425 void
 426 scheduler(void)
 427 {
 428   struct proc *p;
 429   struct cpu *c = mycpu();
 430   
 431   c->proc = 0;
 432   for(;;){
 433     // Avoid deadlock by ensuring that devices can interrupt.
 434     intr_on();
 435 
 436     for(p = proc; p < &proc[NPROC]; p++) {
 437       acquire(&p->lock);
 438       if(p->state == RUNNABLE) {
 439         // Switch to chosen process.  It is the process's job
 440         // to release its lock and then reacquire it
 441         // before jumping back to us.
 442         p->state = RUNNING;
 443         c->proc = p;
 444         swtch(&c->scheduler, &p->context);
 445 
 446         // Process is done running for now.
 447         // It should have changed its p->state before coming back.
 448         c->proc = 0;
 449       }
 450       release(&p->lock);
 451     }
 452   }
 453 }

 455 // Switch to scheduler.  Must hold only p->lock
 456 // and have changed proc->state. Saves and restores
 457 // intena because intena is a property of this
 458 // kernel thread, not this CPU. It should
 459 // be proc->intena and proc->noff, but that would
 460 // break in the few places where a lock is held but
 461 // there's no process.
 462 void
 463 sched(void)
 464 {
 465   int intena;
 466   struct proc *p = myproc();
 467 
 468   if(!holding(&p->lock))
 469     panic("sched p->lock");
 470   if(mycpu()->noff != 1)
 471     panic("sched locks");
 472   if(p->state == RUNNING)
 473     panic("sched running");
 474   if(intr_get())
 475     panic("sched interruptible");
 476 
 477   intena = mycpu()->intena;
 478   swtch(&p->context, &mycpu()->scheduler);
 479   mycpu()->intena = intena;
 480 }

 493 // A fork child's very first scheduling by scheduler()
 494 // will swtch to forkret.
 495 void
 496 forkret(void)
 497 {
 498   static int first = 1;
 499 
 500   // Still holding p->lock from scheduler.
 501   release(&myproc()->lock);
 502 
 503   if (first) {
 504     // File system initialization must be run in the context of a
 505     // regular process (e.g., because it calls sleep), and thus cannot
 506     // be run from main().
 507     first = 0;
 508     fsinit(ROOTDEV);
 509   }
 510 
 511   usertrapret();
 512 }

 514 // Atomically release lock and sleep on chan.
 515 // Reacquires lock when awakened.
 516 void
 517 sleep(void *chan, struct spinlock *lk)
 518 {
 519   struct proc *p = myproc();
 520   
 521   // Must acquire p->lock in order to
 522   // change p->state and then call sched.
 523   // Once we hold p->lock, we can be
 524   // guaranteed that we won't miss any wakeup
 525   // (wakeup locks p->lock),
 526   // so it's okay to release lk.
 527   if(lk != &p->lock){  //DOC: sleeplock0
 528     acquire(&p->lock);  //DOC: sleeplock1
 529     release(lk);
 530   }
 531 
 532   // Go to sleep.
 533   p->chan = chan;
 534   p->state = SLEEPING;
 535 
 536   sched();
 537 
 538   // Tidy up.
 539   p->chan = 0;
 540 
 541   // Reacquire original lock.
 542   if(lk != &p->lock){
 543     release(&p->lock);
 544     acquire(lk);
 545   }
 546 }

 548 // Wake up all processes sleeping on chan.
 549 // Must be called without any p->lock.
 550 void
 551 wakeup(void *chan)
 552 {
 553   struct proc *p;
 554 
 555   for(p = proc; p < &proc[NPROC]; p++) {
 556     acquire(&p->lock);
 557     if(p->state == SLEEPING && p->chan == chan) {
 558       p->state = RUNNABLE;
 559     }
 560     release(&p->lock);
 561   }
 562 }

 574 // Kill the process with the given pid.
 575 // The victim won't exit until it tries to return
 576 // to user space (see usertrap() in trap.c).
 577 int
 578 kill(int pid)
 579 {
 580   struct proc *p;
 581 
 582   for(p = proc; p < &proc[NPROC]; p++){
 583     acquire(&p->lock);
 584     if(p->pid == pid){
 585       p->killed = 1;
 586       if(p->state == SLEEPING){
 587         // Wake process from sleep().
 588         p->state = RUNNABLE;
 589       }
 590       release(&p->lock);
 591       return 0;
 592     }
 593     release(&p->lock);
 594   }
 595   return -1;
 596 }

 628 // Print a process listing to console.  For debugging.
 629 // Runs when user types ^P on console.
 630 // No lock to avoid wedging a stuck machine further.
 631 void
 632 procdump(void)
 633 {
 634   static char *states[] = {
 635   [UNUSED]    "unused",
 636   [SLEEPING]  "sleep ",
 637   [RUNNABLE]  "runble",
 638   [RUNNING]   "run   ",
 639   [ZOMBIE]    "zombie"
 640   };
 641   struct proc *p;
 642   char *state;
 643 
 644   printf("\n");
 645   for(p = proc; p < &proc[NPROC]; p++){
 646     if(p->state == UNUSED)
 647       continue;
 648     if(p->state >= 0 && p->state < NELEM(states) && states[p->state])
 649       state = states[p->state];
 650     else
 651       state = "???";
 652     printf("%d %s %s", p->pid, state, p->name);
 653     printf("\n");
 654   }
 655 }

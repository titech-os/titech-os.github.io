  88 // Look in the process table for an UNUSED proc.
  89 // If found, initialize state required to run in the kernel,
  90 // and return with p->lock held.
  91 // If there are no free procs, or a memory allocation fails, return 0.
  92 static struct proc*
  93 allocproc(void)
  94 {
  95   struct proc *p;
  96 
  97   for(p = proc; p < &proc[NPROC]; p++) {
  98     acquire(&p->lock);
  99     if(p->state == UNUSED) {
 100       goto found;
 101     } else {
 102       release(&p->lock);
 103     }
 104   }
 105   return 0;
 106 
 107 found:
 108   p->pid = allocpid();
 109 
 110   // Allocate a trapframe page.
 111   if((p->trapframe = (struct trapframe *)kalloc()) == 0){
 112     release(&p->lock);
 113     return 0;
 114   }
 115 
 116   // An empty user page table.
 117   p->pagetable = proc_pagetable(p);
 118   if(p->pagetable == 0){
 119     freeproc(p);
 120     release(&p->lock);
 121     return 0;
 122   }
 123 
 124   // Set up new context to start executing at forkret,
 125   // which returns to user space.
 126   memset(&p->context, 0, sizeof(p->context));
 127   p->context.ra = (uint64)forkret;
 128   p->context.sp = p->kstack + PGSIZE;
 129 
 130   return p;
 131 }

 133 // free a proc structure and the data hanging from it,
 134 // including user pages.
 135 // p->lock must be held.
 136 static void
 137 freeproc(struct proc *p)
 138 {
 139   if(p->trapframe)
 140     kfree((void*)p->trapframe);
 141   p->trapframe = 0;
 142   if(p->pagetable)
 143     proc_freepagetable(p->pagetable, p->sz);
 144   p->pagetable = 0;
 145   p->sz = 0;
 146   p->pid = 0;
 147   p->parent = 0;
 148   p->name[0] = 0;
 149   p->chan = 0;
 150   p->killed = 0;
 151   p->xstate = 0;
 152   p->state = UNUSED;
 153 }

 155 // Create a user page table for a given process,
 156 // with no user memory, but with trampoline pages.
 157 pagetable_t
 158 proc_pagetable(struct proc *p)
 159 {
 160   pagetable_t pagetable;
 161 
 162   // An empty page table.
 163   pagetable = uvmcreate();
 164   if(pagetable == 0)
 165     return 0;
 166 
 167   // map the trampoline code (for system call return)
 168   // at the highest user virtual address.
 169   // only the supervisor uses it, on the way
 170   // to/from user space, so not PTE_U.
 171   if(mappages(pagetable, TRAMPOLINE, PGSIZE,
 172               (uint64)trampoline, PTE_R | PTE_X) < 0){
 173     uvmfree(pagetable, 0);
 174     return 0;
 175   }
 176 
 177   // map the trapframe just below TRAMPOLINE, for trampoline.S.
 178   if(mappages(pagetable, TRAPFRAME, PGSIZE,
 179               (uint64)(p->trapframe), PTE_R | PTE_W) < 0){
 180     uvmunmap(pagetable, TRAMPOLINE, 1, 0);
 181     uvmfree(pagetable, 0);
 182     return 0;
 183   }
 184 
 185   return pagetable;
 186 }

 210 // Set up first user process.
 211 void
 212 userinit(void)
 213 {
 214   struct proc *p;
 215 
 216   p = allocproc();
 217   initproc = p;
 218   
 219   // allocate one user page and copy init's instructions
 220   // and data into it.
 221   uvminit(p->pagetable, initcode, sizeof(initcode));
 222   p->sz = PGSIZE;
 223 
 224   // prepare for the very first "return" from kernel to user.
 225   p->trapframe->epc = 0;      // user program counter
 226   p->trapframe->sp = PGSIZE;  // user stack pointer
 227

 229   p->cwd = namei("/");
 230 
 231   p->state = RUNNABLE;
 232 
 233   release(&p->lock);
 234 }

 236 // Grow or shrink user memory by n bytes.
 237 // Return 0 on success, -1 on failure.
 238 int
 239 growproc(int n)
 240 {
 241   uint sz;
 242   struct proc *p = myproc();
 243 
 244   sz = p->sz;
 245   if(n > 0){
 246     if((sz = uvmalloc(p->pagetable, sz, sz + n)) == 0) {
 247       return -1;
 248     }
 249   } else if(n < 0){
 250     sz = uvmdealloc(p->pagetable, sz, sz + n);
 251   }
 252   p->sz = sz;
 253   return 0;

 256 // Create a new process, copying the parent.
 257 // Sets up child kernel stack to return as if from fork() system call.
 258 int
 259 fork(void)
 260 {
 261   int i, pid;
 262   struct proc *np;
 263   struct proc *p = myproc();
 264 
 265   // Allocate process.
 266   if((np = allocproc()) == 0){
 267     return -1;
 268   }
 269 
 270   // Copy user memory from parent to child.
 271   if(uvmcopy(p->pagetable, np->pagetable, p->sz) < 0){
 272     freeproc(np);
 273     release(&np->lock);
 274     return -1;
 275   }
 276   np->sz = p->sz;
 277 
 278   np->parent = p;
 279 
 280   // copy saved user registers.
 281   *(np->trapframe) = *(p->trapframe);
 282 
 283   // Cause fork to return 0 in the child.
 284   np->trapframe->a0 = 0;
 285 
 286   // increment reference counts on open file descriptors.
 287   for(i = 0; i < NOFILE; i++)
 288     if(p->ofile[i])
 289       np->ofile[i] = filedup(p->ofile[i]);
 290   np->cwd = idup(p->cwd);
 291 
 292   safestrcpy(np->name, p->name, sizeof(p->name));
 293 
 294   pid = np->pid;
 295 
 296   np->state = RUNNABLE;
 297 
 298   release(&np->lock);
 299 
 300   return pid;
 301 }

 303 // Pass p's abandoned children to init.
 304 // Caller must hold p->lock.
 305 void
 306 reparent(struct proc *p)
 307 {
 308   struct proc *pp;
 309 
 310   for(pp = proc; pp < &proc[NPROC]; pp++){
 311     // this code uses pp->parent without holding pp->lock.
 312     // acquiring the lock first could cause a deadlock
 313     // if pp or a child of pp were also in exit()
 314     // and about to try to lock p.
 315     if(pp->parent == p){
 316       // pp->parent can't change between the check and the acquire()
 317       // because only the parent changes it, and we're the parent.
 318       acquire(&pp->lock);
 319       pp->parent = initproc;
 320       // we should wake up init here, but that would require
 321       // initproc->lock, which would be a deadlock, since we hold
 322       // the lock on one of init's children (pp). this is why
 323       // exit() always wakes init (before acquiring any locks).
 324       release(&pp->lock);
 325     }
 326   }
 327 }

 329 // Exit the current process.  Does not return.
 330 // An exited process remains in the zombie state
 331 // until its parent calls wait().
 332 void
 333 exit(int status)
 334 {
 335   struct proc *p = myproc();
 336 
 337   if(p == initproc)
 338     panic("init exiting");
 339 
 340   // Close all open files.
 341   for(int fd = 0; fd < NOFILE; fd++){
 342     if(p->ofile[fd]){
 343       struct file *f = p->ofile[fd];
 344       fileclose(f);
 345       p->ofile[fd] = 0;
 346     }
 347   }
 348 
 349   begin_op();
 350   iput(p->cwd);
 351   end_op();
 352   p->cwd = 0;
 353 
 354   // we might re-parent a child to init. we can't be precise about
 355   // waking up init, since we can't acquire its lock once we've
 356   // acquired any other proc lock. so wake up init whether that's
 357   // necessary or not. init may miss this wakeup, but that seems
 358   // harmless.
 359   acquire(&initproc->lock);
 360   wakeup1(initproc);
 361   release(&initproc->lock);
 362 
 363   // grab a copy of p->parent, to ensure that we unlock the same
 364   // parent we locked. in case our parent gives us away to init while
 365   // we're waiting for the parent lock. we may then race with an
 366   // exiting parent, but the result will be a harmless spurious wakeup
 367   // to a dead or wrong process; proc structs are never re-allocated
 368   // as anything else.
 369   acquire(&p->lock);
 370   struct proc *original_parent = p->parent;
 371   release(&p->lock);
 372   
 373   // we need the parent's lock in order to wake it up from wait().
 374   // the parent-then-child rule says we have to lock it first.
 375   acquire(&original_parent->lock);
 376 
 377   acquire(&p->lock);
 378 
 379   // Give any children to init.
 380   reparent(p);
 381 
 382   // Parent might be sleeping in wait().
 383   wakeup1(original_parent);
 384 
 385   p->xstate = status;
 386   p->state = ZOMBIE;
 387 
 388   release(&original_parent->lock);
 389 
 390   // Jump into the scheduler, never to return.
 391   sched();
 392   panic("zombie exit");
 393 }

 395 // Wait for a child process to exit and return its pid.
 396 // Return -1 if this process has no children.
 397 int
 398 wait(uint64 addr)
 399 {
 400   struct proc *np;
 401   int havekids, pid;
 402   struct proc *p = myproc();
 403 
 404   // hold p->lock for the whole time to avoid lost
 405   // wakeups from a child's exit().
 406   acquire(&p->lock);
 407 
 408   for(;;){
 409     // Scan through table looking for exited children.
 410     havekids = 0;
 411     for(np = proc; np < &proc[NPROC]; np++){
 412       // this code uses np->parent without holding np->lock.
 413       // acquiring the lock first would cause a deadlock,
 414       // since np might be an ancestor, and we already hold p->lock.
 415       if(np->parent == p){
 416         // np->parent can't change between the check and the acquire()
 417         // because only the parent changes it, and we're the parent.
 418         acquire(&np->lock);
 419         havekids = 1;
 420         if(np->state == ZOMBIE){
 421           // Found one.
 422           pid = np->pid;
 423           if(addr != 0 && copyout(p->pagetable, addr, (char *)&np->xstate,
 424                                   sizeof(np->xstate)) < 0) {
 425             release(&np->lock);
 426             release(&p->lock);
 427             return -1;
 428           }
 429           freeproc(np);
 430           release(&np->lock);
 431           release(&p->lock);
 432           return pid;
 433         }
 434         release(&np->lock);
 435       }
 436     }
 437 
 438     // No point waiting if we don't have any children.
 439     if(!havekids || p->killed){
 440       release(&p->lock);
 441       return -1;
 442     }
 443     
 444     // Wait for a child to exit.
 445     sleep(p, &p->lock);  //DOC: wait-sleep
 446   }
 447 }

 449 // Per-CPU process scheduler.
 450 // Each CPU calls scheduler() after setting itself up.
 451 // Scheduler never returns.  It loops, doing:
 452 //  - choose a process to run.
 453 //  - swtch to start running that process.
 454 //  - eventually that process transfers control
 455 //    via swtch back to the scheduler.
 456 void
 457 scheduler(void)
 458 {
 459   struct proc *p;
 460   struct cpu *c = mycpu();
 461   
 462   c->proc = 0;
 463   for(;;){
 464     // Avoid deadlock by ensuring that devices can interrupt.
 465     intr_on();
 466 
 467     for(p = proc; p < &proc[NPROC]; p++) {
 468       acquire(&p->lock);
 469       if(p->state == RUNNABLE) {
 470         // Switch to chosen process.  It is the process's job
 471         // to release its lock and then reacquire it
 472         // before jumping back to us.
 473         p->state = RUNNING;
 474         c->proc = p;
 475         swtch(&c->context, &p->context);
 476 
 477         // Process is done running for now.
 478         // It should have changed its p->state before coming back.
 479         c->proc = 0;
 480       }
 481       release(&p->lock);
 482     }
 483   }
 484 }

 486 // Switch to scheduler.  Must hold only p->lock
 487 // and have changed proc->state. Saves and restores
 488 // intena because intena is a property of this
 489 // kernel thread, not this CPU. It should
 490 // be proc->intena and proc->noff, but that would
 491 // break in the few places where a lock is held but
 492 // there's no process.
 493 void
 494 sched(void)
 495 {
 496   int intena;
 497   struct proc *p = myproc();
 498 
 499   if(!holding(&p->lock))
 500     panic("sched p->lock");
 501   if(mycpu()->noff != 1)
 502     panic("sched locks");
 503   if(p->state == RUNNING)
 504     panic("sched running");
 505   if(intr_get())
 506     panic("sched interruptible");
 507 
 508   intena = mycpu()->intena;
 509   swtch(&p->context, &mycpu()->context);
 510   mycpu()->intena = intena;
 511 }

 524 // A fork child's very first scheduling by scheduler()
 525 // will swtch to forkret.
 526 void
 527 forkret(void)
 528 {
 529   static int first = 1;
 530 
 531   // Still holding p->lock from scheduler.
 532   release(&myproc()->lock);
 533 
 534   if (first) {
 535     // File system initialization must be run in the context of a
 536     // regular process (e.g., because it calls sleep), and thus cannot
 537     // be run from main().
 538     first = 0;
 539     fsinit(ROOTDEV);
 540   }
 541 
 542   usertrapret();
 543 }

 545 // Atomically release lock and sleep on chan.
 546 // Reacquires lock when awakened.
 547 void
 548 sleep(void *chan, struct spinlock *lk)
 549 {
 550   struct proc *p = myproc();
 551   
 552   // Must acquire p->lock in order to
 553   // change p->state and then call sched.
 554   // Once we hold p->lock, we can be
 555   // guaranteed that we won't miss any wakeup
 556   // (wakeup locks p->lock),
 557   // so it's okay to release lk.
 558   if(lk != &p->lock){  //DOC: sleeplock0
 559     acquire(&p->lock);  //DOC: sleeplock1
 560     release(lk);
 561   }
 562 
 563   // Go to sleep.
 564   p->chan = chan;
 565   p->state = SLEEPING;
 566 
 567   sched();
 568 
 569   // Tidy up.
 570   p->chan = 0;
 571 
 572   // Reacquire original lock.
 573   if(lk != &p->lock){
 574     release(&p->lock);
 575     acquire(lk);
 576   }
 577 }

 579 // Wake up all processes sleeping on chan.
 580 // Must be called without any p->lock.
 581 void
 582 wakeup(void *chan)
 583 {
 584   struct proc *p;
 585 
 586   for(p = proc; p < &proc[NPROC]; p++) {
 587     acquire(&p->lock);
 588     if(p->state == SLEEPING && p->chan == chan) {
 589       p->state = RUNNABLE;
 590     }
 591     release(&p->lock);
 592   }

 607 // Kill the process with the given pid.
 608 // The victim won't exit until it tries to return
 609 // to user space (see usertrap() in trap.c).
 610 int
 611 kill(int pid)
 612 {
 613   struct proc *p;
 614 
 615   for(p = proc; p < &proc[NPROC]; p++){
 616     acquire(&p->lock);
 617     if(p->pid == pid){
 618       p->killed = 1;
 619       if(p->state == SLEEPING){
 620         // Wake process from sleep().
 621         p->state = RUNNABLE;
 622       }
 623       release(&p->lock);
 624       return 0;
 625     }
 626     release(&p->lock);
 627   }
 628   return -1;
 629 }

 661 // Print a process listing to console.  For debugging.
 662 // Runs when user types ^P on console.
 663 // No lock to avoid wedging a stuck machine further.
 664 void
 665 procdump(void)
 666 {
 667   static char *states[] = {
 668   [UNUSED]    "unused",
 669   [SLEEPING]  "sleep ",
 670   [RUNNABLE]  "runble",
 671   [RUNNING]   "run   ",
 672   [ZOMBIE]    "zombie"
 673   };
 674   struct proc *p;
 675   char *state;
 676 
 677   printf("\n");
 678   for(p = proc; p < &proc[NPROC]; p++){
 679     if(p->state == UNUSED)
 680       continue;
 681     if(p->state >= 0 && p->state < NELEM(states) && states[p->state])
 682       state = states[p->state];
 683     else
 684       state = "???";
 685     printf("%d %s %s", p->pid, state, p->name);
 686     printf("\n");
 687   }
 688 }

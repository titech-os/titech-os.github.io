  86 // Look in the process table for an UNUSED proc.
  87 // If found, initialize state required to run in the kernel,
  88 // and return with p->lock held.
  89 // If there are no free procs, return 0.
  90 static struct proc*
  91 allocproc(void)
  92 {
  93   struct proc *p;
  94 
  95   for(p = proc; p < &proc[NPROC]; p++) {
  96     acquire(&p->lock);
  97     if(p->state == UNUSED) {
  98       goto found;
  99     } else {
 100       release(&p->lock);
 101     }
 102   }
 103   return 0;
 104 
 105 found:
 106   p->pid = allocpid();
 107 
 108   // Allocate a trapframe page.
 109   if((p->tf = (struct trapframe *)kalloc()) == 0){
 110     release(&p->lock);
 111     return 0;
 112   }
 113 
 114   // An empty user page table.
 115   p->pagetable = proc_pagetable(p);
 116 
 117   // Set up new context to start executing at forkret,
 118   // which returns to user space.
 119   memset(&p->context, 0, sizeof p->context);
 120   p->context.ra = (uint64)forkret;
 121   p->context.sp = p->kstack + PGSIZE;
 122 
 123   return p;
 124 }

 126 // free a proc structure and the data hanging from it,
 127 // including user pages.
 128 // p->lock must be held.
 129 static void
 130 freeproc(struct proc *p)
 131 {
 132   if(p->tf)
 133     kfree((void*)p->tf);
 134   p->tf = 0;
 135   if(p->pagetable)
 136     proc_freepagetable(p->pagetable, p->sz);
 137   p->pagetable = 0;
 138   p->sz = 0;
 139   p->pid = 0;
 140   p->parent = 0;
 141   p->name[0] = 0;
 142   p->chan = 0;
 143   p->killed = 0;
 144   p->xstate = 0;
 145   p->state = UNUSED;

 148 // Create a page table for a given process,
 149 // with no user pages, but with trampoline pages.
 150 pagetable_t
 151 proc_pagetable(struct proc *p)
 152 {
 153   pagetable_t pagetable;
 154 
 155   // An empty page table.
 156   pagetable = uvmcreate();
 157 
 158   // map the trampoline code (for system call return)
 159   // at the highest user virtual address.
 160   // only the supervisor uses it, on the way
 161   // to/from user space, so not PTE_U.
 162   mappages(pagetable, TRAMPOLINE, PGSIZE,
 163            (uint64)trampoline, PTE_R | PTE_X);
 164 
 165   // map the trapframe just below TRAMPOLINE, for trampoline.S.
 166   mappages(pagetable, TRAPFRAME, PGSIZE,
 167            (uint64)(p->tf), PTE_R | PTE_W);
 168 
 169   return pagetable;
 170 }

 195 // Set up first user process.
 196 void
 197 userinit(void)
 198 {
 199   struct proc *p;
 200 
 201   p = allocproc();
 202   initproc = p;
 203   
 204   // allocate one user page and copy init's instructions
 205   // and data into it.
 206   uvminit(p->pagetable, initcode, sizeof(initcode));
 207   p->sz = PGSIZE;
 208 
 209   // prepare for the very first "return" from kernel to user.
 210   p->tf->epc = 0;      // user program counter
 211   p->tf->sp = PGSIZE;  // user stack pointer
 212 
 213   safestrcpy(p->name, "initcode", sizeof(p->name));
 214   p->cwd = namei("/");
 215 
 216   p->state = RUNNABLE;
 217 
 218   release(&p->lock);
 219 }

 221 // Grow or shrink user memory by n bytes.
 222 // Return 0 on success, -1 on failure.
 223 int
 224 growproc(int n)
 225 {
 226   uint sz;
 227   struct proc *p = myproc();
 228 
 229   sz = p->sz;
 230   if(n > 0){
 231     if((sz = uvmalloc(p->pagetable, sz, sz + n)) == 0) {
 232       return -1;
 233     }
 234   } else if(n < 0){
 235     sz = uvmdealloc(p->pagetable, sz, sz + n);
 236   }
 237   p->sz = sz;
 238   return 0;

 241 // Create a new process, copying the parent.
 242 // Sets up child kernel stack to return as if from fork() system call.
 243 int
 244 fork(void)
 245 {
 246   int i, pid;
 247   struct proc *np;
 248   struct proc *p = myproc();
 249 
 250   // Allocate process.
 251   if((np = allocproc()) == 0){
 252     return -1;
 253   }
 254 
 255   // Copy user memory from parent to child.
 256   if(uvmcopy(p->pagetable, np->pagetable, p->sz) < 0){
 257     freeproc(np);
 258     release(&np->lock);
 259     return -1;
 260   }
 261   np->sz = p->sz;
 262 
 263   np->parent = p;
 264 
 265   // copy saved user registers.
 266   *(np->tf) = *(p->tf);
 267 
 268   // Cause fork to return 0 in the child.
 269   np->tf->a0 = 0;
 270 
 271   // increment reference counts on open file descriptors.
 272   for(i = 0; i < NOFILE; i++)
 273     if(p->ofile[i])
 274       np->ofile[i] = filedup(p->ofile[i]);
 275   np->cwd = idup(p->cwd);
 276 
 277   safestrcpy(np->name, p->name, sizeof(p->name));
 278 
 279   pid = np->pid;
 280 
 281   np->state = RUNNABLE;
 282 
 283   release(&np->lock);
 284 
 285   return pid;
 286 }

 288 // Pass p's abandoned children to init.
 289 // Caller must hold p->lock.
 290 void
 291 reparent(struct proc *p)
 292 {
 293   struct proc *pp;
 294 
 295   for(pp = proc; pp < &proc[NPROC]; pp++){
 296     // this code uses pp->parent without holding pp->lock.
 297     // acquiring the lock first could cause a deadlock
 298     // if pp or a child of pp were also in exit()
 299     // and about to try to lock p.
 300     if(pp->parent == p){
 301       // pp->parent can't change between the check and the acquire()
 302       // because only the parent changes it, and we're the parent.
 303       acquire(&pp->lock);
 304       pp->parent = initproc;
 305       // we should wake up init here, but that would require
 306       // initproc->lock, which would be a deadlock, since we hold
 307       // the lock on one of init's children (pp). this is why
 308       // exit() always wakes init (before acquiring any locks).
 309       release(&pp->lock);
 310     }
 311   }
 312 }

 314 // Exit the current process.  Does not return.
 315 // An exited process remains in the zombie state
 316 // until its parent calls wait().
 317 void
 318 exit(int status)
 319 {
 320   struct proc *p = myproc();
 321 
 322   if(p == initproc)
 323     panic("init exiting");
 324 
 325   // Close all open files.
 326   for(int fd = 0; fd < NOFILE; fd++){
 327     if(p->ofile[fd]){
 328       struct file *f = p->ofile[fd];
 329       fileclose(f);
 330       p->ofile[fd] = 0;
 331     }
 332   }
 333 
 334   begin_op();
 335   iput(p->cwd);
 336   end_op();
 337   p->cwd = 0;
 338 
 339   // we might re-parent a child to init. we can't be precise about
 340   // waking up init, since we can't acquire its lock once we've
 341   // acquired any other proc lock. so wake up init whether that's
 342   // necessary or not. init may miss this wakeup, but that seems
 343   // harmless.
 344   acquire(&initproc->lock);
 345   wakeup1(initproc);
 346   release(&initproc->lock);
 347 
 348   // grab a copy of p->parent, to ensure that we unlock the same
 349   // parent we locked. in case our parent gives us away to init while
 350   // we're waiting for the parent lock. we may then race with an
 351   // exiting parent, but the result will be a harmless spurious wakeup
 352   // to a dead or wrong process; proc structs are never re-allocated
 353   // as anything else.
 354   acquire(&p->lock);
 355   struct proc *original_parent = p->parent;
 356   release(&p->lock);
 357   
 358   // we need the parent's lock in order to wake it up from wait().
 359   // the parent-then-child rule says we have to lock it first.
 360   acquire(&original_parent->lock);
 361 
 362   acquire(&p->lock);
 363 
 364   // Give any children to init.
 365   reparent(p);
 366 
 367   // Parent might be sleeping in wait().
 368   wakeup1(original_parent);
 369 
 370   p->xstate = status;
 371   p->state = ZOMBIE;
 372 
 373   release(&original_parent->lock);
 374 
 375   // Jump into the scheduler, never to return.
 376   sched();
 377   panic("zombie exit");
 378 }

 380 // Wait for a child process to exit and return its pid.
 381 // Return -1 if this process has no children.
 382 int
 383 wait(uint64 addr)
 384 {
 385   struct proc *np;
 386   int havekids, pid;
 387   struct proc *p = myproc();
 388 
 389   // hold p->lock for the whole time to avoid lost
 390   // wakeups from a child's exit().
 391   acquire(&p->lock);
 392 
 393   for(;;){
 394     // Scan through table looking for exited children.
 395     havekids = 0;
 396     for(np = proc; np < &proc[NPROC]; np++){
 397       // this code uses np->parent without holding np->lock.
 398       // acquiring the lock first would cause a deadlock,
 399       // since np might be an ancestor, and we already hold p->lock.
 400       if(np->parent == p){
 401         // np->parent can't change between the check and the acquire()
 402         // because only the parent changes it, and we're the parent.
 403         acquire(&np->lock);
 404         havekids = 1;
 405         if(np->state == ZOMBIE){
 406           // Found one.
 407           pid = np->pid;
 408           if(addr != 0 && copyout(p->pagetable, addr, (char *)&np->xstate,
 409                                   sizeof(np->xstate)) < 0) {
 410             release(&np->lock);
 411             release(&p->lock);
 412             return -1;
 413           }
 414           freeproc(np);
 415           release(&np->lock);
 416           release(&p->lock);
 417           return pid;
 418         }
 419         release(&np->lock);
 420       }
 421     }
 422 
 423     // No point waiting if we don't have any children.
 424     if(!havekids || p->killed){
 425       release(&p->lock);
 426       return -1;
 427     }
 428     
 429     // Wait for a child to exit.
 430     sleep(p, &p->lock);  //DOC: wait-sleep
 431   }
 432 }

 434 // Per-CPU process scheduler.
 435 // Each CPU calls scheduler() after setting itself up.
 436 // Scheduler never returns.  It loops, doing:
 437 //  - choose a process to run.
 438 //  - swtch to start running that process.
 439 //  - eventually that process transfers control
 440 //    via swtch back to the scheduler.
 441 void
 442 scheduler(void)
 443 {
 444   struct proc *p;
 445   struct cpu *c = mycpu();
 446   
 447   c->proc = 0;
 448   for(;;){
 449     // Avoid deadlock by ensuring that devices can interrupt.
 450     intr_on();
 451 
 452     for(p = proc; p < &proc[NPROC]; p++) {
 453       acquire(&p->lock);
 454       if(p->state == RUNNABLE) {
 455         // Switch to chosen process.  It is the process's job
 456         // to release its lock and then reacquire it
 457         // before jumping back to us.
 458         p->state = RUNNING;
 459         c->proc = p;
 460         swtch(&c->scheduler, &p->context);
 461 
 462         // Process is done running for now.
 463         // It should have changed its p->state before coming back.
 464         c->proc = 0;
 465       }
 466       release(&p->lock);
 467     }
 468   }
 469 }

 471 // Switch to scheduler.  Must hold only p->lock
 472 // and have changed proc->state. Saves and restores
 473 // intena because intena is a property of this
 474 // kernel thread, not this CPU. It should
 475 // be proc->intena and proc->noff, but that would
 476 // break in the few places where a lock is held but
 477 // there's no process.
 478 void
 479 sched(void)
 480 {
 481   int intena;
 482   struct proc *p = myproc();
 483 
 484   if(!holding(&p->lock))
 485     panic("sched p->lock");
 486   if(mycpu()->noff != 1)
 487     panic("sched locks");
 488   if(p->state == RUNNING)
 489     panic("sched running");
 490   if(intr_get())
 491     panic("sched interruptible");
 492 
 493   intena = mycpu()->intena;
 494   swtch(&p->context, &mycpu()->scheduler);
 495   mycpu()->intena = intena;
 496 }

 509 // A fork child's very first scheduling by scheduler()
 510 // will swtch to forkret.
 511 void
 512 forkret(void)
 513 {
 514   static int first = 1;
 515 
 516   // Still holding p->lock from scheduler.
 517   release(&myproc()->lock);
 518 
 519   if (first) {
 520     // File system initialization must be run in the context of a
 521     // regular process (e.g., because it calls sleep), and thus cannot
 522     // be run from main().
 523     first = 0;
 524     fsinit(ROOTDEV);
 525   }
 526 
 527   usertrapret();
 528 }

 530 // Atomically release lock and sleep on chan.
 531 // Reacquires lock when awakened.
 532 void
 533 sleep(void *chan, struct spinlock *lk)
 534 {
 535   struct proc *p = myproc();
 536   
 537   // Must acquire p->lock in order to
 538   // change p->state and then call sched.
 539   // Once we hold p->lock, we can be
 540   // guaranteed that we won't miss any wakeup
 541   // (wakeup locks p->lock),
 542   // so it's okay to release lk.
 543   if(lk != &p->lock){  //DOC: sleeplock0
 544     acquire(&p->lock);  //DOC: sleeplock1
 545     release(lk);
 546   }
 547 
 548   // Go to sleep.
 549   p->chan = chan;
 550   p->state = SLEEPING;
 551 
 552   sched();
 553 
 554   // Tidy up.
 555   p->chan = 0;
 556 
 557   // Reacquire original lock.
 558   if(lk != &p->lock){
 559     release(&p->lock);
 560     acquire(lk);
 561   }
 562 }

 564 // Wake up all processes sleeping on chan.
 565 // Must be called without any p->lock.
 566 void
 567 wakeup(void *chan)
 568 {
 569   struct proc *p;
 570 
 571   for(p = proc; p < &proc[NPROC]; p++) {
 572     acquire(&p->lock);
 573     if(p->state == SLEEPING && p->chan == chan) {
 574       p->state = RUNNABLE;
 575     }
 576     release(&p->lock);
 577   }

 592 // Kill the process with the given pid.
 593 // The victim won't exit until it tries to return
 594 // to user space (see usertrap() in trap.c).
 595 int
 596 kill(int pid)
 597 {
 598   struct proc *p;
 599 
 600   for(p = proc; p < &proc[NPROC]; p++){
 601     acquire(&p->lock);
 602     if(p->pid == pid){
 603       p->killed = 1;
 604       if(p->state == SLEEPING){
 605         // Wake process from sleep().
 606         p->state = RUNNABLE;
 607       }
 608       release(&p->lock);
 609       return 0;
 610     }
 611     release(&p->lock);
 612   }
 613   return -1;
 614 }

 646 // Print a process listing to console.  For debugging.
 647 // Runs when user types ^P on console.
 648 // No lock to avoid wedging a stuck machine further.
 649 void
 650 procdump(void)
 651 {
 652   static char *states[] = {
 653   [UNUSED]    "unused",
 654   [SLEEPING]  "sleep ",
 655   [RUNNABLE]  "runble",
 656   [RUNNING]   "run   ",
 657   [ZOMBIE]    "zombie"
 658   };
 659   struct proc *p;
 660   char *state;
 661 
 662   printf("\n");
 663   for(p = proc; p < &proc[NPROC]; p++){
 664     if(p->state == UNUSED)
 665       continue;
 666     if(p->state >= 0 && p->state < NELEM(states) && states[p->state])
 667       state = states[p->state];
 668     else
 669       state = "???";
 670     printf("%d %s %s", p->pid, state, p->name);
 671     printf("\n");
 672   }
 673 }

 100 // Look in the process table for an UNUSED proc.
 101 // If found, initialize state required to run in the kernel,
 102 // and return with p->lock held.
 103 // If there are no free procs, or a memory allocation fails, return 0.
 104 static struct proc*
 105 allocproc(void)
 106 {
 107   struct proc *p;
 108 
 109   for(p = proc; p < &proc[NPROC]; p++) {
 110     acquire(&p->lock);
 111     if(p->state == UNUSED) {
 112       goto found;
 113     } else {
 114       release(&p->lock);
 115     }
 116   }
 117   return 0;
 118 
 119 found:
 120   p->pid = allocpid();
 121   p->state = USED;
 122 
 123   // Allocate a trapframe page.
 124   if((p->trapframe = (struct trapframe *)kalloc()) == 0){
 125     freeproc(p);
 126     release(&p->lock);
 127     return 0;
 128   }
 129 
 130   // An empty user page table.
 131   p->pagetable = proc_pagetable(p);
 132   if(p->pagetable == 0){
 133     freeproc(p);
 134     release(&p->lock);
 135     return 0;
 136   }
 137 
 138   // Set up new context to start executing at forkret,
 139   // which returns to user space.
 140   memset(&p->context, 0, sizeof(p->context));
 141   p->context.ra = (uint64)forkret;
 142   p->context.sp = p->kstack + PGSIZE;
 143 
 144   return p;
 145 }

 147 // free a proc structure and the data hanging from it,
 148 // including user pages.
 149 // p->lock must be held.
 150 static void
 151 freeproc(struct proc *p)
 152 {
 153   if(p->trapframe)
 154     kfree((void*)p->trapframe);
 155   p->trapframe = 0;
 156   if(p->pagetable)
 157     proc_freepagetable(p->pagetable, p->sz);
 158   p->pagetable = 0;
 159   p->sz = 0;
 160   p->pid = 0;
 161   p->parent = 0;
 162   p->name[0] = 0;
 163   p->chan = 0;
 164   p->killed = 0;
 165   p->xstate = 0;
 166   p->state = UNUSED;
 167 }

 169 // Create a user page table for a given process,
 170 // with no user memory, but with trampoline pages.
 171 pagetable_t
 172 proc_pagetable(struct proc *p)
 173 {
 174   pagetable_t pagetable;
 175 
 176   // An empty page table.
 177   pagetable = uvmcreate();
 178   if(pagetable == 0)
 179     return 0;
 180 
 181   // map the trampoline code (for system call return)
 182   // at the highest user virtual address.
 183   // only the supervisor uses it, on the way
 184   // to/from user space, so not PTE_U.
 185   if(mappages(pagetable, TRAMPOLINE, PGSIZE,
 186               (uint64)trampoline, PTE_R | PTE_X) < 0){
 187     uvmfree(pagetable, 0);
 188     return 0;
 189   }
 190 
 191   // map the trapframe just below TRAMPOLINE, for trampoline.S.
 192   if(mappages(pagetable, TRAPFRAME, PGSIZE,
 193               (uint64)(p->trapframe), PTE_R | PTE_W) < 0){
 194     uvmunmap(pagetable, TRAMPOLINE, 1, 0);
 195     uvmfree(pagetable, 0);
 196     return 0;
 197   }
 198 
 199   return pagetable;
 200 }

 224 // Set up first user process.
 225 void
 226 userinit(void)
 227 {
 228   struct proc *p;
 229 
 230   p = allocproc();
 231   initproc = p;
 232   
 233   // allocate one user page and copy init's instructions
 234   // and data into it.
 235   uvminit(p->pagetable, initcode, sizeof(initcode));
 236   p->sz = PGSIZE;
 237 
 238   // prepare for the very first "return" from kernel to user.
 239   p->trapframe->epc = 0;      // user program counter
 240   p->trapframe->sp = PGSIZE;  // user stack pointer
 241 
 242   safestrcpy(p->name, "initcode", sizeof(p->name));
 243   p->cwd = namei("/");
 244 
 245   p->state = RUNNABLE;
 246 
 247   release(&p->lock);
 248 }

 250 // Grow or shrink user memory by n bytes.
 251 // Return 0 on success, -1 on failure.
 252 int
 253 growproc(int n)
 254 {
 255   uint sz;
 256   struct proc *p = myproc();
 257 
 258   sz = p->sz;
 259   if(n > 0){
 260     if((sz = uvmalloc(p->pagetable, sz, sz + n)) == 0) {
 261       return -1;
 262     }
 263   } else if(n < 0){
 264     sz = uvmdealloc(p->pagetable, sz, sz + n);
 265   }
 266   p->sz = sz;
 267   return 0;

 270 // Create a new process, copying the parent.
 271 // Sets up child kernel stack to return as if from fork() system call.
 272 int
 273 fork(void)
 274 {
 275   int i, pid;
 276   struct proc *np;
 277   struct proc *p = myproc();
 278 
 279   // Allocate process.
 280   if((np = allocproc()) == 0){
 281     return -1;
 282   }
 283 
 284   // Copy user memory from parent to child.
 285   if(uvmcopy(p->pagetable, np->pagetable, p->sz) < 0){
 286     freeproc(np);
 287     release(&np->lock);
 288     return -1;
 289   }
 290   np->sz = p->sz;
 291 
 292   // copy saved user registers.
 293   *(np->trapframe) = *(p->trapframe);
 294 
 295   // Cause fork to return 0 in the child.
 296   np->trapframe->a0 = 0;
 297 
 298   // increment reference counts on open file descriptors.
 299   for(i = 0; i < NOFILE; i++)
 300     if(p->ofile[i])
 301       np->ofile[i] = filedup(p->ofile[i]);
 302   np->cwd = idup(p->cwd);
 303 
 304   safestrcpy(np->name, p->name, sizeof(p->name));
 305 
 306   pid = np->pid;
 307 
 308   release(&np->lock);
 309 
 310   acquire(&wait_lock);
 311   np->parent = p;
 312   release(&wait_lock);
 313 
 314   acquire(&np->lock);
 315   np->state = RUNNABLE;
 316   release(&np->lock);
 317 
 318   return pid;
 319 }

 336 // Exit the current process.  Does not return.
 337 // An exited process remains in the zombie state
 338 // until its parent calls wait().
 339 void
 340 exit(int status)
 341 {
 342   struct proc *p = myproc();
 343 
 344   if(p == initproc)
 345     panic("init exiting");
 346 
 347   // Close all open files.
 348   for(int fd = 0; fd < NOFILE; fd++){
 349     if(p->ofile[fd]){
 350       struct file *f = p->ofile[fd];
 351       fileclose(f);
 352       p->ofile[fd] = 0;
 353     }
 354   }
 355 
 356   begin_op();
 357   iput(p->cwd);
 358   end_op();
 359   p->cwd = 0;
 360 
 361   acquire(&wait_lock);
 362 
 363   // Give any children to init.
 364   reparent(p);
 365 
 366   // Parent might be sleeping in wait().
 367   wakeup(p->parent);
 368   
 369   acquire(&p->lock);
 370 
 371   p->xstate = status;
 372   p->state = ZOMBIE;
 373 
 374   release(&wait_lock);
 375 
 376   // Jump into the scheduler, never to return.
 377   sched();
 378   panic("zombie exit");
 379 }

 381 // Wait for a child process to exit and return its pid.
 382 // Return -1 if this process has no children.
 383 int
 384 wait(uint64 addr)
 385 {
 386   struct proc *np;
 387   int havekids, pid;
 388   struct proc *p = myproc();
 389 
 390   acquire(&wait_lock);
 391 
 392   for(;;){
 393     // Scan through table looking for exited children.
 394     havekids = 0;
 395     for(np = proc; np < &proc[NPROC]; np++){
 396       if(np->parent == p){
 397         // make sure the child isn't still in exit() or swtch().
 398         acquire(&np->lock);
 399 
 400         havekids = 1;
 401         if(np->state == ZOMBIE){
 402           // Found one.
 403           pid = np->pid;
 404           if(addr != 0 && copyout(p->pagetable, addr, (char *)&np->xstate,
 405                                   sizeof(np->xstate)) < 0) {
 406             release(&np->lock);
 407             release(&wait_lock);
 408             return -1;
 409           }
 410           freeproc(np);
 411           release(&np->lock);
 412           release(&wait_lock);
 413           return pid;
 414         }
 415         release(&np->lock);
 416       }
 417     }
 418 
 419     // No point waiting if we don't have any children.
 420     if(!havekids || p->killed){
 421       release(&wait_lock);
 422       return -1;
 423     }
 424     
 425     // Wait for a child to exit.
 426     sleep(p, &wait_lock);  //DOC: wait-sleep
 427   }
 428 }

 430 // Per-CPU process scheduler.
 431 // Each CPU calls scheduler() after setting itself up.
 432 // Scheduler never returns.  It loops, doing:
 433 //  - choose a process to run.
 434 //  - swtch to start running that process.
 435 //  - eventually that process transfers control
 436 //    via swtch back to the scheduler.
 437 void
 438 scheduler(void)
 439 {
 440   struct proc *p;
 441   struct cpu *c = mycpu();
 442   
 443   c->proc = 0;
 444   for(;;){
 445     // Avoid deadlock by ensuring that devices can interrupt.
 446     intr_on();
 447 
 448     for(p = proc; p < &proc[NPROC]; p++) {
 449       acquire(&p->lock);
 450       if(p->state == RUNNABLE) {
 451         // Switch to chosen process.  It is the process's job
 452         // to release its lock and then reacquire it
 453         // before jumping back to us.
 454         p->state = RUNNING;
 455         c->proc = p;
 456         swtch(&c->context, &p->context);
 457 
 458         // Process is done running for now.
 459         // It should have changed its p->state before coming back.
 460         c->proc = 0;
 461       }
 462       release(&p->lock);
 463     }
 464   }
 465 }

 467 // Switch to scheduler.  Must hold only p->lock
 468 // and have changed proc->state. Saves and restores
 469 // intena because intena is a property of this
 470 // kernel thread, not this CPU. It should
 471 // be proc->intena and proc->noff, but that would
 472 // break in the few places where a lock is held but
 473 // there's no process.
 474 void
 475 sched(void)
 476 {
 477   int intena;
 478   struct proc *p = myproc();
 479 
 480   if(!holding(&p->lock))
 481     panic("sched p->lock");
 482   if(mycpu()->noff != 1)
 483     panic("sched locks");
 484   if(p->state == RUNNING)
 485     panic("sched running");
 486   if(intr_get())
 487     panic("sched interruptible");
 488 
 489   intena = mycpu()->intena;
 490   swtch(&p->context, &mycpu()->context);
 491   mycpu()->intena = intena;
 492 }

 505 // A fork child's very first scheduling by scheduler()
 506 // will swtch to forkret.
 507 void
 508 forkret(void)
 509 {
 510   static int first = 1;
 511 
 512   // Still holding p->lock from scheduler.
 513   release(&myproc()->lock);
 514 
 515   if (first) {
 516     // File system initialization must be run in the context of a
 517     // regular process (e.g., because it calls sleep), and thus cannot
 518     // be run from main().
 519     first = 0;
 520     fsinit(ROOTDEV);
 521   }
 522 
 523   usertrapret();
 524 }

 526 // Atomically release lock and sleep on chan.
 527 // Reacquires lock when awakened.
 528 void
 529 sleep(void *chan, struct spinlock *lk)
 530 {
 531   struct proc *p = myproc();
 532   
 533   // Must acquire p->lock in order to
 534   // change p->state and then call sched.
 535   // Once we hold p->lock, we can be
 536   // guaranteed that we won't miss any wakeup
 537   // (wakeup locks p->lock),
 538   // so it's okay to release lk.
 539 
 540   acquire(&p->lock);  //DOC: sleeplock1
 541   release(lk);
 542 
 543   // Go to sleep.
 544   p->chan = chan;
 545   p->state = SLEEPING;
 546 
 547   sched();
 548 
 549   // Tidy up.
 550   p->chan = 0;
 551 
 552   // Reacquire original lock.
 553   release(&p->lock);
 554   acquire(lk);
 555 }

 557 // Wake up all processes sleeping on chan.
 558 // Must be called without any p->lock.
 559 void
 560 wakeup(void *chan)
 561 {
 562   struct proc *p;
 563 
 564   for(p = proc; p < &proc[NPROC]; p++) {
 565     if(p != myproc()){
 566       acquire(&p->lock);
 567       if(p->state == SLEEPING && p->chan == chan) {
 568         p->state = RUNNABLE;
 569       }
 570       release(&p->lock);
 571     }
 572   }
 573 }

 575 // Kill the process with the given pid.
 576 // The victim won't exit until it tries to return
 577 // to user space (see usertrap() in trap.c).
 578 int
 579 kill(int pid)
 580 {
 581   struct proc *p;
 582 
 583   for(p = proc; p < &proc[NPROC]; p++){
 584     acquire(&p->lock);
 585     if(p->pid == pid){
 586       p->killed = 1;
 587       if(p->state == SLEEPING){
 588         // Wake process from sleep().
 589         p->state = RUNNABLE;
 590       }
 591       release(&p->lock);
 592       return 0;
 593     }
 594     release(&p->lock);
 595   }
 596   return -1;
 597 }

 629 // Print a process listing to console.  For debugging.
 630 // Runs when user types ^P on console.
 631 // No lock to avoid wedging a stuck machine further.
 632 void
 633 procdump(void)
 634 {
 635   static char *states[] = {
 636   [UNUSED]    "unused",
 637   [SLEEPING]  "sleep ",
 638   [RUNNABLE]  "runble",
 639   [RUNNING]   "run   ",
 640   [ZOMBIE]    "zombie"
 641   };
 642   struct proc *p;
 643   char *state;
 644 
 645   printf("\n");
 646   for(p = proc; p < &proc[NPROC]; p++){
 647     if(p->state == UNUSED)
 648       continue;
 649     if(p->state >= 0 && p->state < NELEM(states) && states[p->state])
 650       state = states[p->state];
 651     else
 652       state = "???";
 653     printf("%d %s %s", p->pid, state, p->name);
 654     printf("\n");
 655   }
 656 }

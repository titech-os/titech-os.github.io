  18 // the UART control registers.

  33 void
  34 uartinit(void)
  35 {
  36   // disable interrupts.
  37   WriteReg(IER, 0x00);
  38 
  39   // special mode to set baud rate.
  40   WriteReg(LCR, 0x80);
  41 
  42   // LSB for baud rate of 38.4K.
  43   WriteReg(0, 0x03);
  44 
  45   // MSB for baud rate of 38.4K.
  46   WriteReg(1, 0x00);
  47 
  48   // leave set-baud mode,
  49   // and set word length to 8 bits, no parity.
  50   WriteReg(LCR, 0x03);
  51 
  52   // reset and enable FIFOs.
  53   WriteReg(FCR, 0x07);
  54 
  55   // enable receive interrupts.
  56   WriteReg(IER, 0x01);
  57 }

  59 // write one output character to the UART.
  60 void
  61 uartputc(int c)
  62 {
  63   // wait for Transmit Holding Empty to be set in LSR.
  64   while((ReadReg(LSR) & (1 << 5)) == 0)
  65     ;
  66   WriteReg(THR, c);
  67 }

  69 // read one input character from the UART.
  70 // return -1 if none is waiting.
  71 int
  72 uartgetc(void)
  73 {
  74   if(ReadReg(LSR) & 0x01){
  75     // input data is ready.
  76     return ReadReg(RHR);
  77   } else {
  78     return -1;

  80 }
